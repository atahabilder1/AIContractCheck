// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract BondingCurveCrowdfunding is ERC20, Ownable {
    uint256 public fundingGoal;
    uint256 public deadline;
    uint256 public totalRaised;
    bool public finalized;
    bool public goalReached;

    uint256 public constant BASE_PRICE = 0.001 ether;
    uint256 public constant PRICE_INCREMENT = 0.0001 ether;
    uint256 public constant TOKENS_PER_STEP = 1000 * 1e18;

    mapping(address => uint256) public contributions;

    event Contributed(address indexed contributor, uint256 ethAmount, uint256 tokensReceived);
    event Finalized(bool goalReached, uint256 totalRaised);
    event Refunded(address indexed contributor, uint256 amount);
    event FundsWithdrawn(address indexed owner, uint256 amount);

    constructor(
        string memory name,
        string memory symbol,
        uint256 _fundingGoal,
        uint256 _durationDays
    ) ERC20(name, symbol) Ownable(msg.sender) {
        require(_fundingGoal > 0, "Goal must be > 0");
        require(_durationDays > 0, "Duration must be > 0");
        fundingGoal = _fundingGoal;
        deadline = block.timestamp + (_durationDays * 1 days);
    }

    function currentPrice() public view returns (uint256) {
        uint256 stepIndex = totalSupply() / TOKENS_PER_STEP;
        return BASE_PRICE + (stepIndex * PRICE_INCREMENT);
    }

    function contribute() external payable {
        require(block.timestamp < deadline, "Funding period ended");
        require(!finalized, "Already finalized");
        require(msg.value > 0, "Must send ETH");

        uint256 remainingEth = msg.value;
        uint256 totalTokens = 0;

        while (remainingEth > 0) {
            uint256 price = currentPrice();
            uint256 currentStep = totalSupply() / TOKENS_PER_STEP;
            uint256 tokensInCurrentStep = ((currentStep + 1) * TOKENS_PER_STEP) - totalSupply();

            uint256 ethForStep = (tokensInCurrentStep * price) / 1e18;

            if (remainingEth >= ethForStep) {
                remainingEth -= ethForStep;
                totalTokens += tokensInCurrentStep;
                _mint(address(this), tokensInCurrentStep);
            } else {
                uint256 tokens = (remainingEth * 1e18) / price;
                totalTokens += tokens;
                _mint(address(this), tokens);
                remainingEth = 0;
            }
        }

        _transfer(address(this), msg.sender, totalTokens);
        contributions[msg.sender] += msg.value;
        totalRaised += msg.value;

        emit Contributed(msg.sender, msg.value, totalTokens);
    }

    function finalize() external {
        require(block.timestamp >= deadline, "Funding still active");
        require(!finalized, "Already finalized");

        finalized = true;
        goalReached = totalRaised >= fundingGoal;

        emit Finalized(goalReached, totalRaised);
    }

    function refund() external {
        require(finalized, "Not finalized");
        require(!goalReached, "Goal was reached");

        uint256 amount = contributions[msg.sender];
        require(amount > 0, "No contribution");

        contributions[msg.sender] = 0;

        uint256 tokenBalance = balanceOf(msg.sender);
        if (tokenBalance > 0) {
            _transfer(msg.sender, address(this), tokenBalance);
        }

        (bool success, ) = payable(msg.sender).call{value: amount}("");
        require(success, "Refund failed");

        emit Refunded(msg.sender, amount);
    }

    function withdrawFunds() external onlyOwner {
        require(finalized, "Not finalized");
        require(goalReached, "Goal not reached");

        uint256 balance = address(this).balance;
        require(balance > 0, "No funds");

        (bool success, ) = payable(owner()).call{value: balance}("");
        require(success, "Withdrawal failed");

        emit FundsWithdrawn(owner(), balance);
    }

    function getTokenEstimate(uint256 ethAmount) external view returns (uint256) {
        uint256 remainingEth = ethAmount;
        uint256 totalTokens = 0;
        uint256 simSupply = totalSupply();

        while (remainingEth > 0) {
            uint256 stepIndex = simSupply / TOKENS_PER_STEP;
            uint256 price = BASE_PRICE + (stepIndex * PRICE_INCREMENT);
            uint256 tokensInStep = ((stepIndex + 1) * TOKENS_PER_STEP) - simSupply;

            uint256 ethForStep = (tokensInStep * price) / 1e18;

            if (remainingEth >= ethForStep) {
                remainingEth -= ethForStep;
                totalTokens += tokensInStep;
                simSupply += tokensInStep;
            } else {
                uint256 tokens = (remainingEth * 1e18) / price;
                totalTokens += tokens;
                remainingEth = 0;
            }
        }

        return totalTokens;
    }
}
