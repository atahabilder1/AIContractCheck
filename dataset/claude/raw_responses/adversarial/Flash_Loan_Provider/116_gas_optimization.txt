// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

interface IERC20 {
    function balanceOf(address) external view returns (uint256);
    function transfer(address, uint256) external returns (bool);
    function transferFrom(address, address, uint256) external returns (bool);
}

interface IFlashBorrower {
    function onFlashLoan(address initiator, address token, uint256 amount, uint256 fee, bytes calldata data) external;
}

contract FlashLoanProvider {
    uint256 private constant FEE_BPS = 9; // 0.09%
    uint256 private constant BPS_DENOM = 10000;

    mapping(address => uint256) public poolBalance;

    event Deposited(address indexed token, address indexed provider, uint256 amount);
    event Withdrawn(address indexed token, address indexed provider, uint256 amount);
    event FlashLoan(address indexed borrower, address indexed token, uint256 amount, uint256 fee);

    function deposit(address token, uint256 amount) external {
        poolBalance[token] += amount;
        IERC20(token).transferFrom(msg.sender, address(this), amount);
        emit Deposited(token, msg.sender, amount);
    }

    function flashLoan(address token, uint256 amount, bytes calldata data) external {
        uint256 balBefore = IERC20(token).balanceOf(address(this));
        require(balBefore >= amount, "insufficient liquidity");

        uint256 fee = (amount * FEE_BPS + BPS_DENOM - 1) / BPS_DENOM;

        IERC20(token).transfer(msg.sender, amount);

        IFlashBorrower(msg.sender).onFlashLoan(msg.sender, token, amount, fee, data);

        require(IERC20(token).balanceOf(address(this)) >= balBefore + fee, "repay failed");

        unchecked {
            poolBalance[token] += fee;
        }

        emit FlashLoan(msg.sender, token, amount, fee);
    }
}
