pragma solidity ^0.8.0;

interface IFlashLoanReceiver {
    function executeOperation(
        address sender,
        address token,
        uint256 amount,
        uint256 fee,
        bytes calldata params
    ) external;
}

contract FlashLoanProvider {
    mapping(address => uint256) public balances;

    function deposit(address token, uint256 amount) external {
        require(amount > 0, "Amount must be greater than 0");
        balances[token] += amount;
    }

    function flashLoan(address receiverAddress, address token, uint256 amount, bytes memory params) external {
        uint256 fee = amount * 5 / 1000; // 0.5% fee
        require(balances[token] >= amount + fee, "Insufficient balance");

        balances[token] -= amount;
        IFlashLoanReceiver(receiverAddress).executeOperation(msg.sender, token, amount, fee, params);
        require(balances[token] >= amount + fee, "Flash loan not paid back");
        balances[token] += fee;
    }
}
