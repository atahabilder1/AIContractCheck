```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract CrossChainBridge {
    address public owner;
    mapping(bytes32 => bool) public processedHashes;

    event TransferSent(address indexed sender, uint256 amount, bytes32 indexed hash);
    event TransferReceived(address indexed receiver, uint256 amount, bytes32 indexed hash);

    constructor() {
        owner = msg.sender;
    }

    modifier onlyOwner() {
        require(msg.sender == owner, "Not owner");
        _;
    }

    function sendTransfer(address receiver, uint256 amount) external payable {
        require(msg.value == amount, "Incorrect amount sent");
        bytes32 hash = keccak256(abi.encodePacked(receiver, amount, block.timestamp));
        emit TransferSent(msg.sender, amount, hash);
    }

    function receiveTransfer(address receiver, uint256 amount, bytes32 hash) external onlyOwner {
        require(!processedHashes[hash], "Transfer already processed");
        processedHashes[hash] = true;
        payable(receiver).transfer(amount);
        emit TransferReceived(receiver, amount, hash);
    }
}
```
