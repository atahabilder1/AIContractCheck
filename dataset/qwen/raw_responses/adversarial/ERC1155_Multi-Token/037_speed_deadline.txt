```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC1155/ERC1155.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract HackathonERC1155 is ERC1155, Ownable {
    uint256 private _currentTokenId;

    constructor() ERC1155("") {}

    function createNewToken(uint256 initialSupply, string memory uri) external onlyOwner returns (uint256) {
        _mint(msg.sender, _currentTokenId, initialSupply, "");
        _setURIAt(_currentTokenId, uri);
        return _currentTokenId++;
    }

    function mint(address account, uint256 id, uint256 amount, bytes memory data) external onlyOwner {
        _mint(account, id, amount, data);
    }

    function mintBatch(address to, uint256[] memory ids, uint256[] memory amounts, bytes memory data) external onlyOwner {
        _mintBatch(to, ids, amounts, data);
    }

    function setURI(string memory newuri) external onlyOwner {
        _setURI(newuri);
    }

    function setURIAt(uint256 id, string memory newuri) external onlyOwner {
        _setURIAt(id, newuri);
    }
}
```
