```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC1155/ERC1155.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract CustomERC1155 is ERC1155, Ownable {
    mapping(uint256 => uint256) public tokenSupplyCap;
    mapping(uint256 => uint256) public tokenRoyalties;
    mapping(uint256 => uint256) public mintedSupply;

    constructor(string memory uri) ERC1155(uri) {}

    function setTokenSupplyCap(uint256 id, uint256 cap) external onlyOwner {
        tokenSupplyCap[id] = cap;
    }

    function setTokenRoyalty(uint256 id, uint256 royalty) external onlyOwner {
        require(royalty <= 10000, "Royalty cannot exceed 10000 (100%)");
        tokenRoyalties[id] = royalty;
    }

    function mintBatch(
        address to,
        uint256[] memory ids,
        uint256[] memory amounts
    ) external onlyOwner {
        require(ids.length == amounts.length, "Ids and amounts length mismatch");

        for (uint256 i = 0; i < ids.length; i++) {
            uint256 id = ids[i];
            uint256 amount = amounts[i];
            require(
                mintedSupply[id] + amount <= tokenSupplyCap[id],
                "Minting would exceed supply cap"
            );
            mintedSupply[id] += amount;
        }

        _mintBatch(to, ids, amounts, "");
    }

    function uri(uint256 _id) public view override returns (string memory) {
        return string(abi.encodePacked(super.uri(_id), Strings.toString(_id)));
    }
}
```
