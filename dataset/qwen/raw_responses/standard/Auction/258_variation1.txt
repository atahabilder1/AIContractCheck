// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract DutchAuction {
    address public seller;
    uint256 public startingPrice;
    uint256 public auctionStart;
    uint256 public priceDecayRate; // Price decay per second
    uint256 public reservePrice;
    address public winner;

    event AuctionWon(address winner, uint256 price);

    constructor(uint256 _startingPrice, uint256 _priceDecayRate, uint256 _reservePrice) {
        require(_startingPrice > _reservePrice, "Starting price must be greater than reserve price");
        seller = msg.sender;
        startingPrice = _startingPrice;
        priceDecayRate = _priceDecayRate;
        reservePrice = _reservePrice;
        auctionStart = block.timestamp;
    }

    function getCurrentPrice() public view returns (uint256) {
        uint256 timeElapsed = block.timestamp - auctionStart;
        uint256 decayedAmount = timeElapsed * priceDecayRate;
        uint256 currentPrice = startingPrice - decayedAmount;
        return currentPrice > reservePrice ? currentPrice : reservePrice;
    }

    function bid() public payable {
        require(winner == address(0), "Auction has already been won");
        uint256 currentPrice = getCurrentPrice();
        require(msg.value >= currentPrice, "Bid amount is lower than the current price");

        if (msg.value > currentPrice) {
            payable(msg.sender).transfer(msg.value - currentPrice);
        }

        winner = msg.sender;
        payable(seller).transfer(currentPrice);

        emit AuctionWon(msg.sender, currentPrice);
    }

    function getAuctionDetails() public view returns (uint256, uint256, uint256, address) {
        return (getCurrentPrice(), auctionStart, priceDecayRate, winner);
    }
}
