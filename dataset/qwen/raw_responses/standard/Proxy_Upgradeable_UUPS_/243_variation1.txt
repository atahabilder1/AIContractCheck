```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/proxy/utils/Initializable.sol";
import "@openzeppelin/contracts/access/OwnableUpgradeable.sol";
import "@openzeppelin/contracts/proxy/utils/UUPSUpgradeable.sol";

contract MultiApproverUUPS is Initializable, OwnableUpgradeable, UUPSUpgradeable {
    address public securityCouncil;

    modifier requireApproval() {
        require(msg.sender == owner() && msg.sender == securityCouncil, "Approval required from both owner and security council");
        _;
    }

    function initialize(address _securityCouncil) initializer public {
        __Ownable_init();
        __UUPSUpgradeable_init();
        securityCouncil = _securityCouncil;
    }

    function _authorizeUpgrade(address newImplementation) internal override requireApproval {}
}
```
