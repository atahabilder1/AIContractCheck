```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/proxy/ERC1967/ERC1967Upgrade.sol";
import "@openzeppelin/contracts/proxy/utils/UUPSUpgradeable.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract UUPSProxyWithCouncil is UUPSUpgradeable, Ownable {
    address public securityCouncil;
    address public pendingImplementation;

    event UpgradeProposed(address indexed newImplementation);
    event UpgradeApproved(address indexed newImplementation);

    constructor(address _securityCouncil) {
        require(_securityCouncil != address(0), "Security council address cannot be zero");
        securityCouncil = _securityCouncil;
    }

    modifier onlyCouncil() {
        require(msg.sender == securityCouncil, "Caller is not the security council");
        _;
    }

    function proposeUpgrade(address newImplementation) external onlyOwner {
        require(newImplementation != address(0), "New implementation address cannot be zero");
        pendingImplementation = newImplementation;
        emit UpgradeProposed(newImplementation);
    }

    function approveUpgrade() external onlyCouncil {
        require(pendingImplementation != address(0), "No upgrade proposed");
        _upgradeTo(pendingImplementation);
        emit UpgradeApproved(pendingImplementation);
        pendingImplementation = address(0);
    }

    function _authorizeUpgrade(address newImplementation) internal override onlyOwner {
        require(pendingImplementation == newImplementation, "Upgrade not approved by council");
    }
}
```
