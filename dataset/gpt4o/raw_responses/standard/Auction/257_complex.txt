```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC721/IERC721.sol";
import "@openzeppelin/contracts/token/ERC721/IERC721Receiver.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/utils/Counters.sol";
import "@openzeppelin/contracts/utils/math/SafeMath.sol";

contract NFTAuctionHouse is IERC721Receiver, Ownable {
    using Counters for Counters.Counter;
    using SafeMath for uint256;

    enum AuctionType { English, SealedBid, Batch }
    enum AuctionState { Created, Started, Ended, Cancelled }

    struct Auction {
        uint256 auctionId;
        IERC721 nftContract;
        uint256 tokenId;
        address payable seller;
        uint256 reservePrice;
        uint256 startTime;
        uint256 endTime;
        uint256 royaltyPercentage; // e.g., 5 for 5%
        address payable creator;
        AuctionType auctionType;
        AuctionState auctionState;
    }

    struct Bid {
        address payable bidder;
        uint256 bidAmount;
    }

    Counters.Counter private auctionCounter;
    mapping(uint256 => Auction) public auctions;
    mapping(uint256 => Bid[]) public bids;
    uint256 public antiSnipingTime = 300; // 5 minutes

    event AuctionCreated(uint256 auctionId, address indexed nftContract, uint256 tokenId, uint256 reservePrice, uint256 startTime, uint256 endTime, AuctionType auctionType);
    event BidPlaced(uint256 auctionId, address indexed bidder, uint256 bidAmount);
    event AuctionEnded(uint256 auctionId, address winner, uint256 winningBid);
    event AuctionCancelled(uint256 auctionId);

    function createAuction(
        address _nftContract,
        uint256 _tokenId,
        uint256 _reservePrice,
        uint256 _startTime,
        uint256 _endTime,
        uint256 _royaltyPercentage,
        address payable _creator,
        AuctionType _auctionType
    ) external {
        require(_startTime < _endTime, "Start time must be before end time");
        require(_reservePrice > 0, "Reserve price must be greater than zero");

        auctionCounter.increment();
        uint256 auctionId = auctionCounter.current();

        auctions[auctionId] = Auction({
            auctionId: auctionId,
            nftContract: IERC721(_nftContract),
            tokenId: _tokenId,
            seller: payable(msg.sender),
            reservePrice: _reservePrice,
            startTime: _startTime,
            endTime: _endTime,
            royaltyPercentage: _royaltyPercentage,
            creator: _creator,
            auctionType: _auctionType,
            auctionState: AuctionState.Created
        });

        IERC721(_nftContract).transferFrom(msg.sender, address(this), _tokenId);

        emit AuctionCreated(auctionId, _nftContract, _tokenId, _reservePrice, _startTime, _endTime, _auctionType);
    }

    function placeBid(uint256 _auctionId) external payable {
        Auction storage auction = auctions[_auctionId];
        require(block.timestamp >= auction.startTime, "Auction has not started");
        require(block.timestamp <= auction.endTime, "Auction has ended");
        require(auction.auctionState == AuctionState.Started, "Auction is not active");
        require(msg.value > 0, "Bid amount must be greater than zero");

        if (auction.auctionType == AuctionType.English) {
            if (bids[_auctionId].length > 0) {
                Bid memory lastBid = bids[_auctionId][bids[_auctionId].length - 1];
                require(msg.value > lastBid.bidAmount, "Bid must be higher than the last bid");
            }
        }

        bids[_auctionId].push(Bid({
            bidder: payable(msg.sender),
            bidAmount: msg.value
        }));

        if (block.timestamp + antiSnipingTime > auction.endTime) {
            auction.endTime = auction.endTime.add(antiSnipingTime);
        }

        emit BidPlaced(_auctionId, msg.sender, msg.value);
    }

    function endAuction(uint256 _auctionId) external {
        Auction storage auction = auctions[_auctionId];
        require(block.timestamp > auction.endTime, "Auction has not ended yet");
        require(auction.auctionState == AuctionState.Started, "Auction is not active");

        auction.auctionState = AuctionState.Ended;

        if (bids[_auctionId].length > 0) {
            Bid memory highestBid = _determineWinner(_auctionId);

            uint256 royaltyPayment = highestBid.bidAmount.mul(auction.royaltyPercentage).div(100);
            uint256 sellerPayment = highestBid.bidAmount.sub(royaltyPayment);

            auction.creator.transfer(royaltyPayment);
            auction.seller.transfer(sellerPayment);

            auction.nftContract.safeTransferFrom(address(this), highestBid.bidder, auction.tokenId);

            emit AuctionEnded(_auctionId, highestBid.bidder, highestBid.bidAmount);
        } else {
            auction.nftContract.safeTransferFrom(address(this), auction.seller, auction.tokenId);
        }
    }

    function cancelAuction(uint256 _auctionId) external {
        Auction storage auction = auctions[_auctionId];
        require(auction.seller == msg.sender || msg.sender == owner(), "Only seller or owner can cancel");
        require(auction.auctionState == AuctionState.Created || auction.auctionState == AuctionState.Started, "Auction cannot be cancelled");

        auction.auctionState = AuctionState.Cancelled;
        auction.nftContract.safeTransferFrom(address(this), auction.seller, auction.tokenId);

        emit AuctionCancelled(_auctionId);
    }

    function _determineWinner(uint256 _auctionId) internal view returns (Bid memory) {
        Bid memory winningBid;
        uint256 highestBid = 0;

        for (uint256 i = 0; i < bids[_auctionId].length; i++) {
            if (bids[_auctionId][i].bidAmount > highestBid) {
                highestBid = bids[_auctionId][i].bidAmount;
                winningBid = bids[_auctionId][i];
            }
        }

        return winningBid;
    }

    function onERC721Received(address, address, uint256, bytes calldata) external pure override returns (bytes4) {
        return IERC721Receiver.onERC721Received.selector;
    }
}
```
