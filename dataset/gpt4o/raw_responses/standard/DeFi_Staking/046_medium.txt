pragma solidity ^0.8.0;

contract StakingPool {
    struct Stake {
        uint256 amount;
        uint256 startTime;
        uint256 lockUpPeriod;
    }

    mapping(address => Stake) public stakes;
    uint256 public totalStaked;
    address public owner;
    uint256 public constant PENALTY_RATE = 10; // 10% penalty for early withdrawal

    uint256 public constant TIER1_RATE = 5; // 5% reward for 1 month
    uint256 public constant TIER2_RATE = 10; // 10% reward for 3 months
    uint256 public constant TIER3_RATE = 20; // 20% reward for 6 months

    constructor() {
        owner = msg.sender;
    }

    modifier onlyOwner() {
        require(msg.sender == owner, "Not the contract owner");
        _;
    }

    function stake(uint256 lockUpPeriod) external payable {
        require(msg.value > 0, "Stake amount should be greater than zero");
        require(stakes[msg.sender].amount == 0, "Already staked");

        stakes[msg.sender] = Stake({
            amount: msg.value,
            startTime: block.timestamp,
            lockUpPeriod: lockUpPeriod
        });

        totalStaked += msg.value;
    }

    function calculateReward(uint256 amount, uint256 lockUpPeriod) internal pure returns (uint256) {
        if (lockUpPeriod == 30 days) {
            return (amount * TIER1_RATE) / 100;
        } else if (lockUpPeriod == 90 days) {
            return (amount * TIER2_RATE) / 100;
        } else if (lockUpPeriod == 180 days) {
            return (amount * TIER3_RATE) / 100;
        } else {
            return 0;
        }
    }

    function withdraw() external {
        Stake memory userStake = stakes[msg.sender];
        require(userStake.amount > 0, "No active stake");

        uint256 lockUpEnd = userStake.startTime + userStake.lockUpPeriod;
        uint256 amountWithdrawn = userStake.amount;

        if (block.timestamp < lockUpEnd) {
            uint256 penalty = (amountWithdrawn * PENALTY_RATE) / 100;
            amountWithdrawn -= penalty;
        } else {
            uint256 reward = calculateReward(userStake.amount, userStake.lockUpPeriod);
            amountWithdrawn += reward;
        }

        totalStaked -= userStake.amount;
        delete stakes[msg.sender];
        payable(msg.sender).transfer(amountWithdrawn);
    }

    function getStakeDetails(address staker) external view returns (uint256, uint256, uint256) {
        Stake memory userStake = stakes[staker];
        return (userStake.amount, userStake.startTime, userStake.lockUpPeriod);
    }

    function emergencyWithdraw() external onlyOwner {
        payable(owner).transfer(address(this).balance);
    }
}
