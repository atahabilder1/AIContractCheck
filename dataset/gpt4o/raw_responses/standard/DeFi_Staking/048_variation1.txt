```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract StakingContract is Ownable {
    IERC20 public stakingToken;
    ERC20 public stToken;
    
    mapping(address => uint256) public stakedBalances;
    uint256 public totalStaked;

    constructor(IERC20 _stakingToken, string memory name, string memory symbol) {
        stakingToken = _stakingToken;
        stToken = new ERC20(name, symbol);
    }

    function stake(uint256 amount) external {
        require(amount > 0, "Amount must be greater than 0");
        stakingToken.transferFrom(msg.sender, address(this), amount);

        if (totalStaked > 0) {
            uint256 mintAmount = (amount * stToken.totalSupply()) / totalStaked;
            stToken.mint(msg.sender, mintAmount);
        } else {
            stToken.mint(msg.sender, amount);
        }

        stakedBalances[msg.sender] += amount;
        totalStaked += amount;
    }

    function unstake(uint256 stAmount) external {
        require(stAmount > 0, "Amount must be greater than 0");
        require(stToken.balanceOf(msg.sender) >= stAmount, "Insufficient stToken balance");

        uint256 withdrawAmount = (stAmount * totalStaked) / stToken.totalSupply();
        stToken.burnFrom(msg.sender, stAmount);

        stakedBalances[msg.sender] -= withdrawAmount;
        totalStaked -= withdrawAmount;

        stakingToken.transfer(msg.sender, withdrawAmount);
    }
}

contract StakingReceiptToken is ERC20, Ownable {
    constructor(string memory name, string memory symbol) ERC20(name, symbol) {}

    function mint(address to, uint256 amount) external onlyOwner {
        _mint(to, amount);
    }

    function burnFrom(address account, uint256 amount) external onlyOwner {
        _burn(account, amount);
    }
}
```
