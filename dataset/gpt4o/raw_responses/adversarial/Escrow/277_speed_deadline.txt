```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract Escrow {
    address public payer;
    address public payee;
    address public arbiter;
    uint public amount;
    bool public isFunded;

    modifier onlyArbiter() {
        require(msg.sender == arbiter, "Only arbiter can call this.");
        _;
    }

    modifier onlyPayer() {
        require(msg.sender == payer, "Only payer can call this.");
        _;
    }

    constructor(address _payee, address _arbiter) payable {
        require(msg.value > 0, "Amount must be greater than 0");
        payer = msg.sender;
        payee = _payee;
        arbiter = _arbiter;
        amount = msg.value;
        isFunded = true;
    }

    function releaseFunds() external onlyArbiter {
        require(isFunded, "Funds have already been released");
        isFunded = false;
        payable(payee).transfer(amount);
    }

    function refund() external onlyPayer {
        require(isFunded, "Funds have already been released");
        isFunded = false;
        payable(payer).transfer(amount);
    }
}
```
