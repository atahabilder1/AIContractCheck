// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract Staking {
    struct Staker {
        uint256 balance;
        uint256 rewardDebt;
    }

    IERC20 public immutable stakingToken;
    uint256 public immutable rewardPerBlock;
    uint256 public lastRewardBlock;
    uint256 public accRewardPerShare;
    uint256 private immutable precisionFactor;

    mapping(address => Staker) public stakers;

    constructor(IERC20 _stakingToken, uint256 _rewardPerBlock) {
        stakingToken = _stakingToken;
        rewardPerBlock = _rewardPerBlock;
        lastRewardBlock = block.number;
        precisionFactor = 1e12;
    }

    function updatePool() internal {
        if (block.number <= lastRewardBlock) return;
        uint256 stakedSupply = stakingToken.balanceOf(address(this));
        if (stakedSupply == 0) {
            lastRewardBlock = block.number;
            return;
        }
        uint256 blocks = block.number - lastRewardBlock;
        uint256 reward = blocks * rewardPerBlock;
        accRewardPerShare += (reward * precisionFactor) / stakedSupply;
        lastRewardBlock = block.number;
    }

    function deposit(uint256 _amount) external {
        Staker storage user = stakers[msg.sender];
        updatePool();
        if (user.balance > 0) {
            uint256 pending = (user.balance * accRewardPerShare) / precisionFactor - user.rewardDebt;
            if (pending > 0) {
                safeTransferReward(msg.sender, pending);
            }
        }
        if (_amount > 0) {
            stakingToken.transferFrom(msg.sender, address(this), _amount);
            user.balance += _amount;
        }
        user.rewardDebt = (user.balance * accRewardPerShare) / precisionFactor;
    }

    function withdraw(uint256 _amount) external {
        Staker storage user = stakers[msg.sender];
        require(user.balance >= _amount, "Withdraw: Not enough balance");
        updatePool();
        uint256 pending = (user.balance * accRewardPerShare) / precisionFactor - user.rewardDebt;
        if (pending > 0) {
            safeTransferReward(msg.sender, pending);
        }
        if (_amount > 0) {
            user.balance -= _amount;
            stakingToken.transfer(msg.sender, _amount);
        }
        user.rewardDebt = (user.balance * accRewardPerShare) / precisionFactor;
    }

    function safeTransferReward(address _to, uint256 _amount) internal {
        stakingToken.transfer(_to, _amount);
    }
}

interface IERC20 {
    function transfer(address, uint256) external returns (bool);
    function transferFrom(address, address, uint256) external returns (bool);
    function balanceOf(address) external view returns (uint256);
}
