```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract YieldAggregator {
    address public owner;
    mapping(address => uint256) public balances;
    mapping(address => uint256) public depositTimestamps;
    uint256 public constant interestRate = 5; // 5% per year
    uint256 public constant SECONDS_IN_A_YEAR = 365 * 24 * 60 * 60;

    event Deposited(address indexed user, uint256 amount, uint256 timestamp);
    event Withdrawn(address indexed user, uint256 amount, uint256 timestamp);

    modifier onlyOwner() {
        require(msg.sender == owner, "Not the contract owner");
        _;
    }

    constructor() {
        owner = msg.sender;
    }

    function deposit() external payable {
        require(msg.value > 0, "Deposit amount must be greater than zero");

        if (balances[msg.sender] > 0) {
            uint256 pendingInterest = calculateInterest(msg.sender);
            balances[msg.sender] += pendingInterest;
        }

        balances[msg.sender] += msg.value;
        depositTimestamps[msg.sender] = block.timestamp;

        emit Deposited(msg.sender, msg.value, block.timestamp);
    }

    function withdraw(uint256 _amount) external {
        require(balances[msg.sender] >= _amount, "Insufficient balance");

        uint256 pendingInterest = calculateInterest(msg.sender);
        balances[msg.sender] += pendingInterest;
        balances[msg.sender] -= _amount;
        depositTimestamps[msg.sender] = block.timestamp;

        payable(msg.sender).transfer(_amount);

        emit Withdrawn(msg.sender, _amount, block.timestamp);
    }

    function calculateInterest(address _user) public view returns (uint256) {
        uint256 timeDiff = block.timestamp - depositTimestamps[_user];
        uint256 interest = (balances[_user] * interestRate * timeDiff) / (100 * SECONDS_IN_A_YEAR);
        return interest;
    }

    function getBalance(address _user) external view returns (uint256) {
        uint256 pendingInterest = calculateInterest(_user);
        return balances[_user] + pendingInterest;
    }

    function emergencyWithdraw() external onlyOwner {
        payable(owner).transfer(address(this).balance);
    }

    receive() external payable {
        deposit();
    }
}
```
