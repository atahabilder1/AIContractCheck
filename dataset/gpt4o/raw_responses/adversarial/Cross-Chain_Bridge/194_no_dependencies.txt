```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract SimpleCrossChainBridge {
    address public admin;
    uint256 public chainId;
    uint256 public bridgeFee;
    mapping(uint256 => address) private allowedChains;
    mapping(bytes32 => bool) private processedTransactions;

    event TokensLocked(address indexed user, uint256 amount, uint256 targetChainId, address targetAddress);
    event TokensUnlocked(address indexed user, uint256 amount, bytes32 transactionId);

    modifier onlyAdmin() {
        require(msg.sender == admin, "Only admin can perform this action");
        _;
    }

    constructor(uint256 _chainId, uint256 _bridgeFee) {
        admin = msg.sender;
        chainId = _chainId;
        bridgeFee = _bridgeFee;
    }

    function updateBridgeFee(uint256 _bridgeFee) external onlyAdmin {
        bridgeFee = _bridgeFee;
    }

    function allowChain(uint256 _chainId, address _bridgeAddress) external onlyAdmin {
        allowedChains[_chainId] = _bridgeAddress;
    }

    function disallowChain(uint256 _chainId) external onlyAdmin {
        allowedChains[_chainId] = address(0);
    }

    function lockTokens(uint256 amount, uint256 targetChainId, address targetAddress) external payable {
        require(allowedChains[targetChainId] != address(0), "Target chain not allowed");
        require(msg.value >= bridgeFee, "Insufficient bridge fee");

        emit TokensLocked(msg.sender, amount, targetChainId, targetAddress);
    }

    function unlockTokens(address user, uint256 amount, bytes32 transactionId) external onlyAdmin {
        require(!processedTransactions[transactionId], "Transaction already processed");

        processedTransactions[transactionId] = true;
        payable(user).transfer(amount);

        emit TokensUnlocked(user, amount, transactionId);
    }

    receive() external payable {}
}
```
