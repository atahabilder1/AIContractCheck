 ```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/security/Pausable.sol";
import "@openzeppelin/contracts/utils/structs/EnumerableSet.sol";

contract MyToken is ERC20, Ownable, Pausable {
    using EnumerableSet for EnumerableSet.AddressSet;
    EnumerableSet.AddressSet private _blacklistedAddresses;

    constructor() ERC20("MyToken", "MTK") {}

    function mint(address to, uint256 amount) external onlyOwner {
        _mint(to, amount);
    }

    function pause() external onlyOwner {
        _pause();
    }

    function unpause() external onlyOwner {
        _unpause();
    }

    function blacklistAddress(address addr) external onlyOwner {
        require(!isBlacklisted(addr), "Address is already blacklisted");
        EnumerableSet.add(_blacklistedAddresses, addr);
    }

    function removeBlacklistAddress(address addr) external onlyOwner {
        require(isBlacklisted(addr), "Address is not blacklisted");
        EnumerableSet.remove(_blacklistedAddresses, addr);
    }

    function isBlacklisted(address addr) public view returns (bool) {
        return EnumerableSet.contains(_blacklistedAddresses, addr);
    }

    function burn(uint256 amount) external {
        require(!isBlacklisted(msg.sender), "Blacklisted address");
        _burn(msg.sender, amount);
    }

    function _beforeTokenTransfer(address from, address to, uint256 amount)
        internal
        override(ERC20, Pausable)
    {
        super._beforeTokenTransfer(from, to, amount);
        require(!isBlacklisted(from) && !isBlacklisted(to), "Transfer from or to blacklisted address");
    }
}
```
