 ```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";

contract MyToken is ERC20, Ownable, ReentrancyGuard {
    uint256 public constant HARD_CAP = 1000000 ether;
    uint256 public constant SOFT_CAP = 500000 ether;
    uint256 public constant INITIAL_RATE = 1000; // 1 token = 1000 wei
    uint256 public constant RATE_STEP = 100; // Increment rate by 100 for each subsequent round
    uint256 public constant WHITELIST_TIER_1_MIN = 100 ether;
    uint256 public constant WHITELIST_TIER_2_MIN = 500 ether;
    uint256 public constant WHITELIST_TIER_3_MIN = 1000 ether;
    uint256 public constant WHITELIST_TIER_4_MIN = 2000 ether;

    uint256 public totalRaised;
    uint256 public tokenSold;
    bool public saleActive = true;

    mapping(address => uint256) public contributions;
    mapping(address => bool) public whitelisted;
    mapping(address => uint256) public vestingSchedules;

    uint256 public currentRound = 1;
    uint256 public currentRate = INITIAL_RATE;

    event TokensPurchased(address indexed purchaser, uint256 value, uint256 tokens);
    event WhitelistStatusUpdated(address indexed user, bool status);
    event KYCVerified(address indexed user);

    constructor() ERC20("MyToken", "MTK") {}

    function purchaseTokens(uint256 amount) external nonReentrant {
        require(saleActive, "Sale is not active");
        require(amount > 0, "Amount must be greater than zero");
        require(totalRaised + amount <= HARD_CAP, "Hard cap reached");

        if (totalRaised < SOFT_CAP) {
            require(amount >= 1 ether, "Minimum purchase amount is 1 ether");
        }

        if (whitelisted[msg.sender]) {
            require(amount >= getWhitelistTierMin(whitelisted[msg.sender]), "Below minimum whitelist tier amount");
        }

        uint256 tokens = amount * currentRate;
        require(tokens + tokenSold <= HARD_CAP, "Token sale hard cap reached");

        contributions[msg.sender] += amount;
        totalRaised += amount;
        tokenSold += tokens;

        _mint(msg.sender, tokens);

        emit TokensPurchased(msg.sender, amount, tokens);
    }

    function getWhitelistTierMin(uint256 tier) internal pure returns (uint256) {
        if (tier == 1) return WHITELIST_TIER_1_MIN;
        if (tier == 2) return WHITELIST_TIER_2_MIN;
        if (tier == 3) return WHITELIST_TIER_3_MIN;
        if (tier == 4) return WHITELIST_TIER_4_MIN;
        revert("Invalid whitelist tier");
    }

    function updateWhitelistStatus(address user, bool status) external onlyOwner {
        whitelisted[user] = status;
        emit WhitelistStatusUpdated(user, status);
    }

    function startNextRound() external onlyOwner {
        currentRound++;
        currentRate += RATE_STEP;
    }

    function finalizeSale() external onlyOwner {
        require(totalRaised >= SOFT_CAP || tokenSold >= HARD_CAP, "Sale conditions not met");
        saleActive = false;
    }

    function withdraw() external onlyOwner {
        require(!saleActive, "Sale is still active");
        require(address(this).balance > 0, "No funds to withdraw");
        payable(owner()).transfer(address(this).balance);
    }

    function setVestingSchedule(address user, uint256 vestingEnd, uint256 totalTokens) external onlyOwner {
        require(vestingEnd > block.timestamp, "Vesting end must be in the future");
        vestingSchedules[user] = vestingEnd;
        _mint(user, totalTokens);
    }

    function claimTokens() external {
        require(block.timestamp >= vestingSchedules[msg.sender], "Vesting period not over");
        uint256 tokens = balanceOf(msg.sender);
        require(tokens > 0, "No tokens to claim");
        _burn(msg.sender, tokens);
        payable(msg.sender).transfer(tokens * 1 wei);
    }

    function verifyKYC(address user) external onlyOwner {
        require(!whitelisted[user], "User is already whitelisted");
        whitelisted[user] = true;
        emit KYCVerified(user);
    }

    receive() external payable {}
}
```
