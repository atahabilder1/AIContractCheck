 ```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/token/ERC721/extensions/ERC721Enumerable.sol";
import "@openzeppelin/contracts/token/ERC721/extensions/ERC721URIStorage.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/utils/math/SafeMath.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";

contract NFTMarketplace is ERC721, ERC721Enumerable, ERC721URIStorage, Ownable, ReentrancyGuard {
    using SafeMath for uint256;

    struct Auction {
        uint256 tokenId;
        address payable seller;
        uint256 startingPrice;
        uint256 endTime;
        uint256 highestBid;
        address highestBidder;
        bool ended;
        uint256 royaltyFee;
        address royaltyReceiver;
    }

    uint256 public auctionId;
    mapping(uint256 => Auction) public auctions;
    mapping(uint256 => uint256) public auctionEndTimes;
    mapping(uint256 => uint256) public auctionStartingPrices;
    mapping(uint256 => uint256) public auctionRoyaltyFees;
    mapping(uint256 => address) public auctionRoyaltyReceivers;

    event AuctionCreated(uint256 indexed auctionId, uint256 tokenId, uint256 startingPrice, uint256 endTime, uint256 royaltyFee, address royaltyReceiver);
    event AuctionBid(uint256 indexed auctionId, address bidder, uint256 amount);
    event AuctionEnded(uint256 indexed auctionId, address winner, uint256 amount);

    constructor() ERC721("NFTMarketplace", "NFTM") {}

    function createAuction(uint256 tokenId, uint256 startingPrice, uint256 duration, uint256 royaltyFee, address royaltyReceiver) public nonReentrant {
        require(_exists(tokenId), "Token does not exist");
        require(ownerOf(tokenId) == msg.sender, "Sender must be the owner of the token");

        auctionId++;
        auctions[auctionId] = Auction({
            tokenId: tokenId,
            seller: payable(msg.sender),
            startingPrice: startingPrice,
            endTime: block.timestamp + duration,
            highestBid: 0,
            highestBidder: address(0),
            ended: false,
            royaltyFee: royaltyFee,
            royaltyReceiver: royaltyReceiver
        });

        auctionEndTimes[auctionId] = block.timestamp + duration;
        auctionStartingPrices[auctionId] = startingPrice;
        auctionRoyaltyFees[auctionId] = royaltyFee;
        auctionRoyaltyReceivers[auctionId] = royaltyReceiver;

        _transfer(msg.sender, address(this), tokenId);

        emit AuctionCreated(auctionId, tokenId, startingPrice, block.timestamp + duration, royaltyFee, royaltyReceiver);
    }

    function placeBid(uint256 auctionId) public payable nonReentrant {
        Auction storage auction = auctions[auctionId];
        require(block.timestamp < auction.endTime, "Auction already ended");
        require(msg.value > auction.highestBid, "There already is a higher bid");

        if (auction.highestBid != 0) {
            payable(auction.highestBidder).transfer(auction.highestBid);
        }

        auction.highestBid = msg.value;
        auction.highestBidder = msg.sender;

        emit AuctionBid(auctionId, msg.sender, msg.value);
    }

    function endAuction(uint256 auctionId) public nonReentrant {
        Auction storage auction = auctions[auctionId];
        require(block.timestamp >= auction.endTime, "Auction not yet ended");
        require(!auction.ended, "AuctionEnd already called");

        auction.ended = true;
        if (auction.highestBidder != address(0)) {
            uint256 royaltyAmount = auction.highestBid.mul(auction.royaltyFee).div(100);
            uint256 sellerAmount = auction.highestBid.sub(royaltyAmount);

            payable(auction.seller).transfer(sellerAmount);
            payable(auction.royaltyReceiver).transfer(royaltyAmount);

            _transfer(address(this), auction.highestBidder, auction.tokenId);

            emit AuctionEnded(auctionId, auction.highestBidder, auction.highestBid);
        } else {
            _transfer(address(this), auction.seller, auction.tokenId);
            emit AuctionEnded(auctionId, address(0), 0);
        }
    }

    function cancelAuction(uint256 auctionId) public {
        Auction storage auction = auctions[auctionId];
        require(msg.sender == auction.seller, "Only the seller can cancel the auction");
        require(!auction.ended, "Auction already ended");

        _transfer(address(this), auction.seller, auction.tokenId);
        auction.ended = true;
    }

    function extendAuction(uint256 auctionId, uint256 extension) public {
        Auction storage auction = auctions[auctionId];
        require(block.timestamp < auction.endTime, "Auction already ended");
        require(msg.sender == auction.seller, "Only the seller can extend the auction");

        auction.endTime += extension;
    }

    function getAuctions() public view returns (Auction[] memory) {
        Auction[] memory allAuctions = new Auction[](auctionId);
        for (uint256 i = 1; i <= auctionId; i++) {
            allAuctions[i - 1] = auctions[i];
        }
        return allAuctions;
    }

    function _beforeTokenTransfer(address from, address to, uint256 tokenId) internal override(ERC721, ERC721Enumerable) {
        super._beforeTokenTransfer(from, to, tokenId);
    }

    function supportsInterface(bytes4 interfaceId) public view override(ERC721, ERC721Enumerable) returns (bool) {
        return super.supportsInterface(interfaceId);
    }

    function _burn(uint256 tokenId) internal override(ERC721, ERC721URIStorage) {
        super._burn(tokenId);
    }

    function tokenURI(uint256 tokenId) public view override(ERC721, ERC721URIStorage) returns (string memory) {
        return super.tokenURI(tokenId);
    }
}
```
