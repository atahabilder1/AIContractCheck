// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract SimpleMultisigWallet {
    address[] public owners;
    uint public required;
    mapping(address => bool) public isOwner;

    struct Transaction {
        address to;
        uint value;
        bytes data;
        bool executed;
    }

    Transaction[] public transactions;

    modifier onlyOwner() {
        require(isOwner[msg.sender], "Not an owner");
        _;
    }

    modifier validRequirement(uint _ownerCount, uint _required) {
        require(_ownerCount <= 50 && _required <= _ownerCount && _required != 0 && _ownerCount != 0, "Invalid requirement");
        _;
    }

    event Deposit(address indexed sender, uint amount);
    event Submission(uint indexed txIndex);
    event Confirmation(address indexed owner, uint indexed txIndex);
    event Execution(uint indexed txIndex);
    event ExecutionFailure(uint indexed txIndex);

    constructor(address[] memory _owners, uint _required) validRequirement(_owners.length, _required) {
        for (uint i = 0; i < _owners.length; i++) {
            address owner = _owners[i];
            isOwner[owner] = true;
            owners.push(owner);
        }
        required = _required;
    }

    receive() external payable {
        emit Deposit(msg.sender, msg.value);
    }

    function submitTransaction(address _to, uint _value, bytes memory _data) public onlyOwner {
        uint txIndex = transactions.length;
        transactions.push(Transaction({
            to: _to,
            value: _value,
            data: _data,
            executed: false
        }));
        emit Submission(txIndex);
    }

    function confirmTransaction(uint txIndex) public onlyOwner {
        require(transactions[txIndex].to != address(0), "Invalid transaction");
        require(!transactions[txIndex].executed, "Transaction already executed");
        require(!hasConfirmed(txIndex, msg.sender), "Transaction already confirmed");
        transactions[txIndex].confirmations.push(msg.sender);
        emit Confirmation(msg.sender, txIndex);
    }

    function executeTransaction(uint txIndex) public onlyOwner {
        Transaction storage transaction = transactions[txIndex];
        require(transaction.to != address(0), "Invalid transaction");
        require(!transaction.executed, "Transaction already executed");
        require(hasConfirmed(txIndex, msg.sender), "Transaction not confirmed");
        transaction.executed = true;
        (bool success, ) = transaction.to.call{value: transaction.value}(transaction.data);
        if (success) {
            emit Execution(txIndex);
        } else {
            emit ExecutionFailure(txIndex);
        }
    }

    function hasConfirmed(uint txIndex, address owner) internal view returns (bool) {
        for (uint i = 0; i < transactions[txIndex].confirmations.length; i++) {
            if (transactions[txIndex].confirmations[i] == owner) {
                return true;
            }
        }
        return false;
    }
}
