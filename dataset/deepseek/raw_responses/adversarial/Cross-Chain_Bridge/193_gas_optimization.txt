 ```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

interface IERC20 {
    function transfer(address to, uint256 value) external returns (bool);
    function approve(address spender, uint256 value) external returns (bool);
}

contract CrossChainBridge {
    address public owner;
    mapping(address => bool) public trustedContracts;
    mapping(bytes32 => bool) public processedMessages;

    event Deposit(address indexed token, address indexed to, uint256 amount);
    event Withdrawal(address indexed token, address indexed to, uint256 amount);

    modifier onlyOwner() {
        require(msg.sender == owner, "Not the owner");
        _;
    }

    modifier onlyTrustedContract() {
        require(trustedContracts[msg.sender], "Not a trusted contract");
        _;
    }

    constructor() {
        owner = msg.sender;
    }

    function setTrustedContract(address _contract, bool _isTrusted) external onlyOwner {
        trustedContracts[_contract] = _isTrusted;
    }

    function deposit(address token, uint256 amount) external {
        require(trustedContracts[msg.sender], "Only trusted contracts can deposit");
        IERC20(token).transfer(address(this), amount);
        emit Deposit(token, address(this), amount);
    }

    function withdraw(address token, address to, uint256 amount) external onlyTrustedContract {
        IERC20(token).transfer(to, amount);
        emit Withdrawal(token, to, amount);
    }

    function processMessage(bytes32 messageHash, address token, address to, uint256 amount) external onlyTrustedContract {
        require(!processedMessages[messageHash], "Message already processed");
        processedMessages[messageHash] = true;
        withdraw(token, to, amount);
    }
}
```
