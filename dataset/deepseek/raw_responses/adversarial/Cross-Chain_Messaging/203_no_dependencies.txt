 ```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract CrossChainMessaging {
    struct Message {
        address sender;
        string content;
    }

    mapping(uint256 => Message) public messages;
    uint256 public messageCount;

    event MessageSent(uint256 indexed chainId, address indexed sender, string content);

    function sendMessage(uint256 chainId, string memory content) public {
        messages[messageCount] = Message(msg.sender, content);
        emit MessageSent(chainId, msg.sender, content);
        messageCount++;
    }

    function getMessage(uint256 index) public view returns (address, string memory) {
        require(index < messageCount, "Message does not exist");
        Message storage message = messages[index];
        return (message.sender, message.content);
    }
}
```
