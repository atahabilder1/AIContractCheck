 ```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "https://github.com/aave/aave-v3-core/blob/master/contracts/interfaces/IPool.sol";
import "https://github.com/aave/aave-v3-core/blob/master/contracts/interfaces/IPoolAddressesProvider.sol";
import "https://github.com/aave/aave-v3-core/blob/master/contracts/interfaces/ILendingPool.sol";

contract FlashLoanProvider {
    address public constant AAVE_ADDRESS_PROVIDER = 0x24a42fD28C976A7dF4aE0e62b3FE8BCf9e797f48;
    address public constant DAI_ADDRESS = 0x6B175474E89094C44Da98b954EedeAC495271d0F;

    function flashLoan(address _receiver, uint256 _amount, bytes calldata _params) external {
        IPoolAddressesProvider provider = IPoolAddressesProvider(AAVE_ADDRESS_PROVIDER);
        ILendingPool pool = ILendingPool(provider.getLendingPool());

        pool.flashLoan(
            _receiver,
            DAI_ADDRESS,
            _amount,
            _params
        );
    }
}
```
