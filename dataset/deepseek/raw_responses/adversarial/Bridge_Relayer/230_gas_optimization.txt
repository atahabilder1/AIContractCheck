 ```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

interface IERC20 {
    function transfer(address to, uint256 amount) external returns (bool);
    function approve(address spender, uint256 amount) external returns (bool);
}

contract BridgeRelayer {
    address public owner;
    mapping(address => bool) public approvedTokens;

    event TokenApproved(address indexed token);
    event TokenUnapproved(address indexed token);
    event TokensMigrated(address indexed token, address indexed to, uint256 amount);

    modifier onlyOwner() {
        require(msg.sender == owner, "Only owner can call this function");
        _;
    }

    constructor() {
        owner = msg.sender;
    }

    function approveToken(address token) external onlyOwner {
        approvedTokens[token] = true;
        emit TokenApproved(token);
    }

    function unapproveToken(address token) external onlyOwner {
        approvedTokens[token] = false;
        emit TokenUnapproved(token);
    }

    function transferTokens(address token, address to, uint256 amount) external onlyOwner {
        require(approvedTokens[token], "Token not approved");
        IERC20(token).transfer(to, amount);
        emit TokensMigrated(token, to, amount);
    }
}
```
