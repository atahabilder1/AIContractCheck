pragma solidity ^0.8.0;

contract ERC1155MultiToken {
    mapping(address => mapping(uint256 => uint256)) private _balances;
    mapping(uint256 => string) private _tokenURIs;

    event TransferSingle(address indexed operator, address indexed from, address indexed to, uint256 id, uint256 value);
    event TransferBatch(address indexed operator, address indexed from, address indexed to, uint256[] ids, uint256[] values);
    event URI(string value, uint256 indexed id);

    constructor() {
        _tokenURIs[0] = "https://example.com/tokens/{id}.json";
        // Add more tokens and their URIs as needed
    }

    function balanceOf(address account, uint256 id) public view returns (uint256) {
        return _balances[account][id];
    }

    function uri(uint256 id) public view returns (string memory) {
        require(_tokenURIs[id].length > 0, "URI not found");
        return _tokenURIs[id];
    }

    function safeTransferFrom(address from, address to, uint256 id, uint256 amount, bytes calldata data) public {
        _transferFrom(from, to, id, amount);
        emit TransferSingle(msg.sender, from, to, id, amount);
    }

    function safeBatchTransferFrom(address from, address to, uint256[] calldata ids, uint256[] calldata amounts, bytes calldata data) public {
        require(ids.length == amounts.length, "Array length mismatch");

        for (uint256 i = 0; i < ids.length; i++) {
            _transferFrom(from, to, ids[i], amounts[i]);
        }

        emit TransferBatch(msg.sender, from, to, ids, amounts);
    }

    function _transferFrom(address from, address to, uint256 id, uint256 amount) private {
        require(_balances[from][id] >= amount, "Insufficient balance");

        unchecked {
            _balances[from][id] -= amount;
            _balances[to][id] += amount;
        }
    }
}
