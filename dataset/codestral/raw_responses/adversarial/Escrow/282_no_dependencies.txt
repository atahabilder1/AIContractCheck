 ```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract Escrow {
    address public buyer;
    address public seller;
    uint public amount;
    bool public isLocked = false;

    constructor(address payable _buyer, address payable _seller, uint _amount) {
        buyer = _buyer;
        seller = _seller;
        amount = _amount;
    }

    function deposit() public payable {
        require(!isLocked, "Contract is locked");
        require(msg.sender == buyer, "Only the buyer can deposit");
        require(msg.value == amount, "Deposited amount does not match the required amount");
        isLocked = true;
    }

    function releaseFunds() public {
        require(isLocked, "Contract is not locked");
        require(msg.sender == seller, "Only the seller can release funds");
        seller.transfer(amount);
        isLocked = false;
    }
}
```
