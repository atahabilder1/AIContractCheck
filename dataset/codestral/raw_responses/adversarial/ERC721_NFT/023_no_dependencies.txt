 ```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract MyERC721 {
    string public name = "MyERC721";
    string public symbol = "MYNFT";
    uint256 public totalSupply = 0;

    mapping(uint256 => address) private _tokenOwners;
    mapping(address => uint256) private _balances;
    mapping(uint256 => mapping (string => string)) private _tokenURIs;

    event Transfer(address indexed from, address indexed to, uint256 indexed tokenId);
    event Approval(address indexed owner, address indexed approved, uint256 indexed tokenId);

    constructor() { }

    function balanceOf(address _owner) public view returns (uint256) {
        return _balances[_owner];
    }

    function ownerOf(uint256 _tokenId) public view returns (address) {
        return _tokenOwners[_tokenId];
    }

    function safeTransferFrom(address _from, address _to, uint256 _tokenId) public {
        require(_checkOwner(_from, _tokenId), "Not the owner");
        _transfer(_from, _to, _tokenId);
    }

    function transferFrom(address _from, address _to, uint256 _tokenId) public {
        require(_checkOwner(_from, _tokenId), "Not the owner");
        _transfer(_from, _to, _tokenId);
    }

    function approve(address _approved, uint256 _tokenId) public {
        address owner = _tokenOwners[_tokenId];
        require(owner == msg.sender, "Not the owner");
        emit Approval(msg.sender, _approved, _tokenId);
    }

    function setTokenURI(uint256 _tokenId, string memory _uri) public {
        require(_checkOwner(msg.sender, _tokenId), "Not the owner");
        _setTokenURI(_tokenId, _uri);
    }

    function tokenURI(uint256 _tokenId) public view returns (string memory) {
        return _tokenURIs[_tokenId][""];
    }

    function mint(address _to, uint256 _tokenId, string memory _uri) public {
        require(_to != address(0), "Invalid recipient");
        require(_tokenOwners[_tokenId] == address(0), "Token already exists");

        _balances[_to] += 1;
        totalSupply += 1;
        _transfer(address(0), _to, _tokenId);
        _setTokenURI(_tokenId, _uri);
    }

    function burn(uint256 _tokenId) public {
        address owner = _tokenOwners[_tokenId];
        require(owner == msg.sender || _approvals[msg.sender][_tokenId], "Not the owner or approved");

        _balances[owner] -= 1;
        totalSupply -= 1;
        _transfer(owner, address(0), _tokenId);
    }

    function _checkOwner(address _owner, uint256 _tokenId) private view returns (bool) {
        return _tokenOwners[_tokenId] == _owner || _approvals[msg.sender][_tokenId];
    }

    function _transfer(address _from, address _to, uint256 _tokenId) private {
        if (_from != address(0)) {
            _balances[_from] -= 1;
        }
        if (_to != address(0)) {
            _balances[_to] += 1;
        }
        _tokenOwners[_tokenId] = _to;
        emit Transfer(_from, _to, _tokenId);
    }

    function _setTokenURI(uint256 _tokenId, string memory _uri) private {
        _tokenURIs[_tokenId][""] = _uri;
    }
}
```
