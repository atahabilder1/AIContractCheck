// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract FlashLoanProvider {
    address payable owner;
    mapping(address => uint256) balances;

    constructor() {
        owner = payable(msg.sender);
    }

    function deposit() public payable {
        balances[msg.sender] += msg.value;
    }

    function withdraw(uint256 amount) public {
        require(balances[msg.sender] >= amount, "Insufficient balance");
        balances[msg.sender] -= amount;
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success, "Transfer failed");
    }

    function flashLoan(address payable receiver, uint256 amount) public {
        balances[receiver] += amount;
        (bool success, bytes memory data) = receiver.call{value: amount}(abi.encodeWithSignature("executeFlashLoan()"));
        require(success && abi.decode(data, (uint256)), "Flash loan execution failed");
    }
}
