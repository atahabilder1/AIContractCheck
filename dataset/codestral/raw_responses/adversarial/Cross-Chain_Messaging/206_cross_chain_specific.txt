 ```solidity
pragma solidity ^0.8.0;

contract CrossChainMessenger {
    struct Message {
        address sender;
        uint256 nonce;
        bytes data;
    }

    event NewMessage(address indexed sender, uint256 indexed nonce, bytes data);

    function sendMessage(address _receiverChain, bytes memory _data) public {
        Message memory newMessage = Message({
            sender: msg.sender,
            nonce: msg.nonce,
            data: _data
        });

        emit NewMessage(newMessage.sender, newMessage.nonce, newMessage.data);
    }
}
```
This is a simple Solidity smart contract for cross-chain messaging. It emits an event when a message is sent with the sender's address, nonce (transaction count), and data of the message.
