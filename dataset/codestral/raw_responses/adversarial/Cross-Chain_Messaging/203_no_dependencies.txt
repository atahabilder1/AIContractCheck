pragma solidity ^0.8.0;

contract CrossChainMessenger {
    struct Message {
        address sender;
        bytes data;
        uint256 nonce;
    }

    event NewMessage(address indexed _sender, bytes _data, uint256 _nonce);

    mapping (uint256 => bool) usedNonces;
    uint256 public nonce;

    function sendMessage(address _receiverContract, bytes memory _data) external {
        Message memory newMessage = Message(_msgSender(), _data, nonce);
        emit NewMessage(_msgSender(), _data, nonce);
        nonce++;
    }

    function receiveMessage(address _sender, uint256 _nonce, bytes memory _data) external {
        require(!usedNonces[_nonce], "Message already used");
        usedNonces[_nonce] = true;
        // Process message here
    }
}
