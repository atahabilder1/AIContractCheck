 ```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.17;

contract Relayer {
    mapping(bytes32 => uint) public submissionCounts;
    mapping(bytes32 => bool) public messageValidity;

    function submitMessage(bytes32 _message) external {
        submissionCounts[_message]++;
        if (submissionCounts[_message] >= 2) {
            messageValidity[_message] = true;
        }
    }
}
```
