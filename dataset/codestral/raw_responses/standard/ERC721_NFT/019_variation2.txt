pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/utils/math/SafeMath.sol";

contract DutchAuction is ERC721, Ownable {
    using SafeMath for uint;

    uint public constant INITIAL_PRICE = 0.1 ether; // Initial price of each token
    uint public constant DISCOUNT_INTERVAL = 10 minutes; // Price decrease interval
    uint public constant TOTAL_TOKENS = 100; // Total number of tokens for sale

    uint public remainingTokens;
    uint public auctionEndTime;

    constructor() ERC721("DutchAuction", "DAU") {
        remainingTokens = TOTAL_TOKENS;
        auctionEndTime = block.timestamp + DISCOUNT_INTERVAL * (TOTAL_TOKENS - 1);
    }

    function mint() public payable {
        require(remainingTokens > 0, "Auction is over");
        require(msg.value >= getCurrentPrice(), "Insufficient payment");
        _safeMint(msg.sender, TOTAL_TOKENS - remainingTokens);
        remainingTokens = remainingTokens.sub(1);
    }

    function getCurrentPrice() public view returns (uint) {
        uint elapsedTime = block.timestamp > auctionEndTime ? DISCOUNT_INTERVAL * (TOTAL_TOKENS - 1) : block.timestamp.sub(block.timestamp % DISCOUNT_INTERVAL);
        return INITIAL_PRICE - (INITIAL_PRICE * elapsedTime / (DISCOUNT_INTERVAL * (TOTAL_TOKENS - 1)));
    }
}
