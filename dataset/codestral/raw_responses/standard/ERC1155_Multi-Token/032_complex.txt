 ```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC1155/ERC1155.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/utils/math/SafeMath.sol";

contract CraftingNFT is ERC1155, Ownable {
    using SafeMath for uint256;

    string private baseTokenURI;
    mapping(uint256 => string) private tokenURIs;

    constructor() ERC1155("https://example.com/{id}.json") {}

    function craftToken(uint256[] memory _tokenIds, uint256 _amount, string memory _uri) public onlyOwner {
        for (uint256 i = 0; i < _tokenIds.length; i++) {
            _burn(_msgSender(), _tokenIds[i], balanceOf(_msgSender(), _tokenIds[i]));
        }
        uint256 newTokenId = _mint(_msgSender(), _amount, "", "");
        tokenURIs[newTokenId] = baseTokenURI + _uri;
    }

    function _beforeTokenTransfer(address operator, address from, address to, uint256[] memory ids, uint256[] memory amounts, bytes calldata data) internal override {
        super._beforeTokenTransfer(operator, from, to, ids, amounts, data);
    }

    function _burn(address account, uint256 id, uint256 value) private {
        super._burn(account, id, value);
    }

    function _mint(address to, uint256 amount, string memory uri, bytes memory data) internal override returns (uint256) {
        return super._mint(to, amount, uri, data);
    }

    function uri(uint256 tokenId) public view virtual override returns (string memory) {
        require(_exists(tokenId), "ERC1155: URI query for nonexistent token");
        return tokenURIs[tokenId];
    }
}
```
This is a basic implementation of an ERC1155 contract with the requested features. It includes a function to craft new tokens by burning multiple existing ones, and dynamic metadata through the `uri` function override. However, marketplace integration is not included in this code snippet as it requires additional logic depending on the specifics of your marketplace (e.g., listings, sales).
