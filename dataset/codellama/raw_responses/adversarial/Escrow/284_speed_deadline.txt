pragma solidity ^0.8.0;

contract Escrow {
    address public buyer;
    address public seller;
    address public arbiter;
    uint public amount;
    bool public dispute;

    constructor(address _buyer, address _seller, address _arbiter, uint _amount) public {
        buyer = _buyer;
        seller = _seller;
        arbiter = _arbiter;
        amount = _amount;
        dispute = false;
    }

    function release() public {
        require(msg.sender == buyer, "Only the buyer can release the funds");
        require(dispute == false, "Dispute must be resolved before release");
        seller.transfer(amount);
    }

    function disputeRelease() public {
        require(msg.sender == arbiter, "Only the arbiter can release the funds in case of a dispute");
        require(dispute == true, "Dispute must be raised before release");
        seller.transfer(amount);
    }

    function dispute() public {
        require(msg.sender == buyer || msg.sender == seller, "Only the buyer or seller can raise a dispute");
        dispute = true;
    }

    function resolveDispute() public {
        require(msg.sender == arbiter, "Only the arbiter can resolve the dispute");
        dispute = false;
    }
}
