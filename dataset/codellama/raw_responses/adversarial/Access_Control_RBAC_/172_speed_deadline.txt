pragma solidity ^0.8.0;

contract AccessControl {
    address public owner;
    mapping (address => mapping (string => bool)) public roles;

    constructor() public {
        owner = msg.sender;
    }

    modifier onlyOwner() {
        require(msg.sender == owner, "Only the contract owner can perform this action");
        _;
    }

    modifier onlyWithRole(string memory role) {
        require(roles[msg.sender][role], "Only users with the required role can perform this action");
        _;
    }

    function grantRole(address user, string memory role) public onlyOwner {
        roles[user][role] = true;
    }

    function revokeRole(address user, string memory role) public onlyOwner {
        roles[user][role] = false;
    }
}
