
```

pragma solidity ^0.8.0;

contract RBAC {
    mapping(address => bool) private roles;

    function assignRole(address _addr, string memory _role) public {
        roles[_addr] = true;
    }

    function revokeRole(address _addr, string memory _role) public {
        roles[_addr] = false;
    }

    function hasRole(address _addr, string memory _role) public view returns (bool) {
        return roles[_addr];
    }
}
```
This is a very simple RBAC contract that allows you to assign and revoke roles to addresses. It uses a mapping to store the roles and a function to check if an address has a specific role. The mapping is private, so only the contract can modify it.
