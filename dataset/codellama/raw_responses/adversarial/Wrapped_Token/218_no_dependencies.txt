```
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract WrappedToken {
    // Token name
    string public name;
    // Token symbol
    string public symbol;
    // Token total supply
    uint256 public totalSupply;
    // Mapping of token balances
    mapping(address => uint256) public balances;
    // Mapping of allowances
    mapping(address => mapping(address => uint256)) public allowances;

    constructor(string memory _name, string memory _symbol, uint256 _totalSupply) public {
        name = _name;
        symbol = _symbol;
        totalSupply = _totalSupply;
        balances[msg.sender] = _totalSupply;
    }

    function transfer(address _to, uint256 _value) public returns (bool) {
        require(msg.sender == _to || allowances[msg.sender][_to] >= _value, "Insufficient allowance");
        require(balances[msg.sender] >= _value, "Insufficient balance");
        balances[msg.sender] -= _value;
        balances[_to] += _value;
        allowances[msg.sender][_to] -= _value;
        return true;
    }

    function transferFrom(address _from, address _to, uint256 _value) public returns (bool) {
        require(allowances[_from][msg.sender] >= _value, "Insufficient allowance");
        require(balances[_from] >= _value, "Insufficient balance");
        balances[_from] -= _value;
        balances[_to] += _value;
        allowances[_from][msg.sender] -= _value;
        return true;
    }

    function approve(address _spender, uint256 _value) public returns (bool) {
        allowances[msg.sender][_spender] = _value;
        return true;
    }

    function balanceOf(address _owner) public view returns (uint256) {
        return balances[_owner];
    }
}
```
This Wrapped Token contract has the following functions:

* `transfer`: Allows the owner to transfer tokens to another address.
* `transferFrom`: Allows a spender to transfer tokens from the owner's address to another address.
* `approve`: Allows the owner to approve a spender to transfer tokens from the owner's address.
* `balanceOf`: Returns the balance of tokens held by the owner.

Note that this is a basic implementation of a Wrapped Token contract and may not be suitable for production use. It is important to thoroughly test and audit any smart contract before deploying it on the blockchain.
