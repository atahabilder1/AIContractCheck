```
pragma solidity ^0.8.0;

contract ICO {

    address public owner;
    address public token;
    uint256 public softCap;
    uint256 public hardCap;
    uint256 public startTime;
    uint256 public endTime;
    mapping(address => uint256) public purchases;
    mapping(address => uint256) public vestingSchedules;
    mapping(address => bool) public kycVerified;

    constructor(address _owner, address _token, uint256 _softCap, uint256 _hardCap, uint256 _startTime, uint256 _endTime) public {
        owner = _owner;
        token = _token;
        softCap = _softCap;
        hardCap = _hardCap;
        startTime = _startTime;
        endTime = _endTime;
    }

    function buyTokens(address _buyer, uint256 _amount) public {
        require(msg.sender == owner, "Only the owner can buy tokens");
        require(block.timestamp >= startTime, "ICO has not started yet");
        require(block.timestamp <= endTime, "ICO has already ended");
        require(_amount > 0, "Invalid amount");
        require(purchases[_buyer] + _amount <= hardCap, "Exceeds hard cap");
        require(kycVerified[_buyer], "KYC verification required");

        purchases[_buyer] += _amount;
        vestingSchedules[_buyer] = _amount;

        // Transfer tokens to the buyer
        token.transfer(_buyer, _amount);
    }

    function refund() public {
        require(msg.sender == owner, "Only the owner can refund");
        require(block.timestamp > endTime, "ICO has not ended yet");
        require(purchases[msg.sender] > 0, "No tokens purchased");

        uint256 amount = purchases[msg.sender];
        purchases[msg.sender] = 0;

        // Transfer tokens back to the owner
        token.transfer(msg.sender, amount);
    }

    function verifyKYC(address _buyer) public {
        require(msg.sender == owner, "Only the owner can verify KYC");
        require(!kycVerified[_buyer], "KYC already verified");

        kycVerified[_buyer] = true;
    }

    function getTokensSold() public view returns (uint256) {
        return token.balanceOf(address(this));
    }

    function getSoftCapReached() public view returns (bool) {
        return token.balanceOf(address(this)) >= softCap;
    }

    function getHardCapReached() public view returns (bool) {
        return token.balanceOf(address(this)) >= hardCap;
    }
}
```
This smart contract implements a full ICO/token sale with whitelist tiers, vesting schedules for purchased tokens, soft/hard caps, multiple rounds with different prices, and KYC verification hooks. The contract has the following functions:

* `buyTokens`: Allows the owner to buy tokens for a given amount. The function checks if the ICO has started, if the amount is greater than 0, if the hard cap is not exceeded, if the buyer has KYC verification, and if the owner has enough tokens to cover the purchase. If the purchase is successful, the function updates the purchases mapping, vesting schedules mapping, and transfers the tokens to the buyer.
* `refund`: Allows the owner to refund their tokens if the ICO has not reached the soft cap. The function checks if the ICO has ended, if the owner has tokens to refund, and if the owner has KYC verification. If the refund is successful, the function updates the purchases mapping and transfers the tokens back to the owner.
* `verifyKYC`: Allows the owner to verify the KYC of a buyer. The function checks if the owner is the one calling the function and if the buyer has not been verified yet. If the verification is successful, the function updates the KYC verification mapping.
* `getTokensSold`: Returns the total number of tokens sold during the ICO.
* `getSoftCapReached`: Returns a boolean indicating if the soft cap has been reached.
* `getHardCapReached`: Returns a boolean indicating if the hard cap has been reached.

The contract also has the following variables:

* `owner`: The address of the owner of the ICO.
* `token`: The address of the token being sold.
* `softCap`: The soft cap of the ICO.
* `hardCap`: The hard cap of the ICO.
* `startTime`: The start time of the ICO.
* `endTime`: The end time of the ICO.
* `purchases`: A mapping that stores the amount of tokens purchased by each buyer.
* `vestingSchedules`: A mapping that stores the vesting schedule for each buyer.
* `kycVerified`: A mapping that stores the KYC verification status for each buyer.
