```
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC1155/ERC1155.sol";
import "@openzeppelin/contracts/token/ERC1155/ERC1155Receiver.sol";
import "@openzeppelin/contracts/utils/Context.sol";
import "@openzeppelin/contracts/utils/math/SafeMath.sol";

contract MyNFT is Context, ERC1155 {
    using SafeMath for uint256;

    // Mapping of token IDs to token supply
    mapping(uint256 => uint256) private _tokenSupply;

    // Mapping of token IDs to token metadata
    mapping(uint256 => string) private _tokenMetadata;

    // Mapping of token IDs to token metadata URI
    mapping(uint256 => string) private _tokenMetadataURI;

    constructor() ERC1155("MyNFT") {}

    function mint(
        address to,
        uint256 id,
        uint256 amount,
        bytes memory data
    ) public virtual override {
        require(to != address(0), "ERC1155: mint to the zero address");
        require(amount > 0, "ERC1155: mint amount must be greater than zero");

        // Update token supply
        _tokenSupply[id] = _tokenSupply[id].add(amount);

        // Mint the token
        _mint(to, id, amount, data);
    }

    function burn(
        address from,
        uint256 id,
        uint256 amount
    ) public virtual override {
        require(from != address(0), "ERC1155: burn from the zero address");
        require(amount > 0, "ERC1155: burn amount must be greater than zero");

        // Update token supply
        _tokenSupply[id] = _tokenSupply[id].sub(amount);

        // Burn the token
        _burn(from, id, amount);
    }

    function craft(
        address from,
        uint256[] memory ids,
        uint256[] memory amounts
    ) public virtual override {
        require(from != address(0), "ERC1155: craft from the zero address");

        // Update token supply
        for (uint256 i = 0; i < ids.length; i++) {
            _tokenSupply[ids[i]] = _tokenSupply[ids[i]].add(amounts[i]);
        }

        // Craft the tokens
        _craft(from, ids, amounts);
    }

    function _craft(
        address from,
        uint256[] memory ids,
        uint256[] memory amounts
    ) private {
        // Mint the new token
        mint(from, ids[0], amounts[0], "");

        // Burn the old tokens
        for (uint256 i = 1; i < ids.length; i++) {
            burn(from, ids[i], amounts[i]);
        }
    }

    function setTokenMetadata(uint256 id, string memory metadata) public {
        require(msg.sender == _tokenMetadata[id], "ERC1155: only the token owner can set metadata");

        _tokenMetadata[id] = metadata;
    }

    function setTokenMetadataURI(uint256 id, string memory uri) public {
        require(msg.sender == _tokenMetadataURI[id], "ERC1155: only the token owner can set metadata URI");

        _tokenMetadataURI[id] = uri;
    }

    function tokenMetadata(uint256 id) public view returns (string memory) {
        return _tokenMetadata[id];
    }

    function tokenMetadataURI(uint256 id) public view returns (string memory) {
        return _tokenMetadataURI[id];
    }

    function tokenSupply(uint256 id) public view returns (uint256) {
        return _tokenSupply[id];
    }
}
```
