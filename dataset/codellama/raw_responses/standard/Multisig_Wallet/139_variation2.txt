pragma solidity ^0.8.0;

contract SocialRecoveryWallet {
    address private owner;
    address[] public guardians;

    constructor(address _owner, address[] memory _guardians) {
        owner = _owner;
        guardians = _guardians;
    }

    function recover() public {
        require(msg.sender == owner, "Only the owner can recover the wallet");
        require(guardians.length >= 3, "At least 3 guardians are required to recover the wallet");

        // Collect the signatures from the guardians
        bytes[] memory signatures = new bytes[](guardians.length);
        for (uint256 i = 0; i < guardians.length; i++) {
            signatures[i] = recoverSignature(guardians[i]);
        }

        // Verify the signatures
        require(verifySignatures(signatures), "Invalid signatures");

        // Recover the wallet
        owner = address(0);
        emit Recovered(owner);
    }

    function recoverSignature(address _guardian) private view returns (bytes memory) {
        return keccak256(abi.encodePacked(_guardian));
    }

    function verifySignatures(bytes[] memory _signatures) private view returns (bool) {
        for (uint256 i = 0; i < _signatures.length; i++) {
            if (_signatures[i] != recoverSignature(guardians[i])) {
                return false;
            }
        }

        return true;
    }

    event Recovered(address indexed _owner);
}
