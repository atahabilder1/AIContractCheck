
Here is an example of a Solidity smart contract for an escrow contract for freelance services:
```
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract FreelanceEscrow {
    address public client;
    address public freelancer;
    uint256 public milestoneCount;
    uint256 public totalAmount;
    uint256 public releasedAmount;
    mapping(uint256 => bool) public milestones;

    constructor(address _client, address _freelancer, uint256 _totalAmount) public {
        client = _client;
        freelancer = _freelancer;
        totalAmount = _totalAmount;
        releasedAmount = 0;
        milestoneCount = 0;
    }

    function addMilestone(uint256 _amount) public {
        require(msg.sender == client, "Only the client can add milestones");
        require(_amount > 0, "Invalid amount");
        require(totalAmount >= _amount, "Insufficient balance");
        milestones[milestoneCount] = true;
        milestoneCount++;
        totalAmount -= _amount;
    }

    function releaseMilestone(uint256 _milestoneIndex) public {
        require(msg.sender == client, "Only the client can release milestones");
        require(_milestoneIndex >= 0 && _milestoneIndex < milestoneCount, "Invalid milestone index");
        require(milestones[_milestoneIndex], "Milestone not yet approved");
        uint256 amount = totalAmount * (_milestoneIndex + 1) / milestoneCount;
        releasedAmount += amount;
        totalAmount -= amount;
    }

    function getBalance() public view returns (uint256) {
        return releasedAmount;
    }
}
```
This contract has the following functions:

* `constructor(address _client, address _freelancer, uint256 _totalAmount)`: This is the constructor function that sets up the escrow contract. It takes the addresses of the client and freelancer, and the total amount of the project.
* `addMilestone(uint256 _amount)`: This function allows the client to add a new milestone to the contract. It takes the amount of the milestone as an argument.
* `releaseMilestone(uint256 _milestoneIndex)`: This function allows the client to release a milestone. It takes the index of the milestone as an argument.
* `getBalance()`: This function returns the balance of the contract.

The contract uses a mapping to store the milestones and their corresponding amounts. The `milestones` mapping is a mapping from the milestone index to a boolean value indicating whether the milestone has been approved or not. The `totalAmount` variable stores the total amount of the project, and the `releasedAmount` variable stores the total amount that has been released.

The `addMilestone` function checks that the sender is the client, and that the amount is greater than 0 and less than or equal to the total amount of the project. It then adds the milestone to the `milestones` mapping and increases the `milestoneCount` variable.

The `releaseMilestone` function checks that the sender is the client, and that the milestone index is valid and has been approved. It then releases the amount of the milestone to the freelancer, and increases the `releasedAmount` variable.

The `getBalance` function returns the balance of the contract, which is the amount that has been released.
