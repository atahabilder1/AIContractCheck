```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/token/ERC20/extensions/ERC20Permit.sol";

contract CappedPermitToken is ERC20, ERC20Permit {
    // Define the maximum total supply as 1 billion tokens with 18 decimals
    uint256 public constant MAX_SUPPLY = 1_000_000_000 * (10**18);

    constructor(string memory name, string memory symbol)
        ERC20(name, symbol)
        ERC20Permit(name) // ERC20Permit requires the token name
    {
        // Mint the entire capped supply to the deployer of the contract
        _mint(msg.sender, MAX_SUPPLY);
    }
}
```
