        require(msg.sender == owner, "Caller is not the owner");
        _;
    }

    modifier onlyValidator() {
        require(validators[msg.sender].isValidator, "Caller is not a validator");
        _;
    }

    constructor(address _messageBus) {
        messageBus = _messageBus;
        owner = msg.sender;
    }

    function addValidator(address _validatorAddress) public onlyOwner {
        require(!validators[_validatorAddress].isValidator, "Already a validator");
        validators[_validatorAddress].isValidator = true;
        emit ValidatorAdded(_validatorAddress);
    }

    function removeValidator(address _validatorAddress) public onlyOwner {
        require(validators[_validatorAddress].isValidator, "Not a validator");
        validators[_validatorAddress].isValidator = false;
        // If validator has stake, they will need to unstake it
        if (validators[_validatorAddress].stake > 0) {
            unstakeRequests[_validatorAddress] = block.timestamp;
        }
        emit ValidatorRemoved(_validatorAddress);
    }

    function stake() public payable onlyValidator {
        require(msg.value >= MIN_STAKE_AMOUNT, "Minimum stake not met");
        validators[msg.sender].stake += msg.value;
        validators[msg.sender].lastStakedTimestamp = block.timestamp;
        emit StakeAdded(msg.sender, msg.value);
    }

    function unstake() public onlyValidator {
        uint256 availableStake = validators[msg.sender].stake;
        require(availableStake > 0, "No stake to unstake");

        // Prevent unstaking if the validator has been removed and is awaiting delay
        if (unstakeRequests[msg.sender] > 0) {
            require(block.timestamp >= unstakeRequests[msg.sender] + UNSTAKE_DELAY, "Unstake delay period not met");
        }

        validators[msg.sender].stake = 0;
        emit StakeRemoved(msg.sender, availableStake);
        (bool success, ) = payable(msg.sender).call{value: availableStake}("");
        require(success, "Unstake transfer failed");
    }

    function requestUnstake() public onlyValidator {
        require(validators[msg.sender].stake > 0, "No stake to request unstake");
        require(unstakeRequests[msg.sender] == 0, "Unstake already requested");
        unstakeRequests[msg.sender] = block.timestamp;
        emit UnstakeRequested(msg.sender, block.timestamp);
    }

    function claimUnstake() public onlyValidator {
        uint256 requestTimestamp = unstakeRequests[msg.sender];
        require(requestTimestamp > 0, "No unstake request found");
        require(block.timestamp >= requestTimestamp + UNSTAKE_DELAY, "Unstake delay period not met");

        uint256 amountToClaim = validators[msg.sender].stake;
        require(amountToClaim > 0, "No stake available to claim");

        unstakeRequests[msg.sender] = 0; // Clear the request
        validators[msg.sender].stake = 0; // Remove stake

        emit StakeRemoved(msg.sender, amountToClaim);
        emit UnstakeClaimed(msg.sender, amountToClaim);

        (bool success, ) = payable(msg.sender).call{value: amountToClaim}("");
        require(success, "Unstake claim transfer failed");
    }
