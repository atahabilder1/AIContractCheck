// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract Bridge is Ownable {

    address public immutable feeRecipient;
    uint256 public immutable erc20FeePercentage; // Fee percentage for ERC20 transfers (e.g., 100 for 1%)
    uint256 public immutable nativeFeePercentage; // Fee percentage for native ETH transfers (e.g., 100 for 1%)

    event ERC20TransferInitiated(address indexed from, address indexed to, address indexed token, uint256 amount, uint256 fee);
    event NativeTransferInitiated(address indexed from, address indexed to, uint256 amount, uint256 fee);

    constructor(
        address _feeRecipient,
        uint256 _erc20FeePercentage,
        uint256 _nativeFeePercentage
    ) {
        require(_feeRecipient != address(0), "Fee recipient cannot be zero address");
        feeRecipient = _feeRecipient;
        erc20FeePercentage = _erc20FeePercentage;
        nativeFeePercentage = _nativeFeePercentage;
    }

    /**
     * @notice Initiates a cross-chain transfer of an ERC20 token.
     * @dev The sender must approve this contract to spend the specified token amount.
     * @param _token The address of the ERC20 token to transfer.
     * @param _to The recipient address on the destination chain.
     * @param _amount The amount of tokens to transfer.
     */
    function initiateERC20Transfer(
        address _token,
        address _to,
        uint256 _amount
    ) external {
        require(_token != address(0), "Token address cannot be zero address");
        require(_to != address(0), "Recipient address cannot be zero address");
        require(_amount > 0, "Amount must be greater than zero");

        IERC20 token = IERC20(_token);
        uint256 fee = (_amount * erc20FeePercentage) / 10000; // 10000 for 100%
        uint256 amountAfterFee = _amount - fee;

        require(token.balanceOf(msg.sender) >= _amount, "Insufficient token balance");
        require(token.transferFrom(msg.sender, address(this), _amount), "ERC20 transfer failed");

        // In a real cross-chain scenario, this would involve sending a message
        // to the destination chain to mint/transfer the tokens.
        // For this example, we'll simulate by emitting an event and sending fees.
        emit ERC20TransferInitiated(msg.sender, _to, _token, amountAfterFee, fee);

        if (fee > 0) {
            // Transfer fees to the fee recipient. This is a simplified example.
            // In a real bridge, fees might be handled differently (e.g., collected on destination chain).
            require(token.transfer(feeRecipient, fee), "Fee transfer failed");
        }
    }

    /**
     * @notice Initiates a cross-chain transfer of native ETH.
     * @param _to The recipient address on the destination chain.
     * @param _amount The amount of ETH to transfer.
     */
    function initiateNativeTransfer(
        address _to
    ) external payable {
        require(_to != address(0), "Recipient address cannot be zero address");
        require(msg.value > 0, "Amount must be greater than zero");

        uint256 fee = (msg.value * nativeFeePercentage) / 10000; // 10000 for 100%
        uint256 amountAfterFee = msg.value - fee;

        // In a real cross-chain scenario, this would involve sending a message
        // to the destination chain to mint/transfer the ETH.
        // For this example, we'll simulate by emitting an event and sending fees.
        emit NativeTransferInitiated(msg.sender, _to, amountAfterFee, fee);

        if (fee > 0) {
            // Transfer fees to the fee recipient.
            (bool success, ) = payable(feeRecipient).call{value: fee}("");
            require(success, "Native fee transfer failed");
        }

        // In a real bridge, the `amountAfterFee` would be sent to the destination chain.
        // Here, we'll just return it to the sender as a simulation of the remaining balance.
        // This is NOT how a real bridge would work.
        if (amountAfterFee > 0) {
             (bool success, ) = payable(msg.sender).call{value: amountAfterFee}("");
             require(success, "Return of non-fee amount failed");
        }
    }

    // --- Admin functions ---

    /**
     * @notice Allows the owner to withdraw any accidentally sent ETH to this contract.
     */
    function withdrawStuckETH() external onlyOwner {
        (bool success, ) = payable(owner()).call{value: address(this).balance}("");
        require(success, "Stuck ETH withdrawal failed");
    }

    /**
     * @notice Allows the owner to withdraw any accidentally sent ERC20 tokens to this contract.
     * @param _token The address of the ERC20 token to withdraw.
     */
    function withdrawStuckERC20(address _token) external onlyOwner {
        require(_token != address(0), "Token address cannot be zero address");
        IERC20 token = IERC20(_token);
        uint256 balance = token.balanceOf(address(this));
        if (balance > 0) {
            require(token.transfer(owner(), balance), "Stuck ERC20 withdrawal failed");
        }
    }

    // Fallback function to receive ETH directly, though initiateNativeTransfer is preferred.
    receive() external payable {}
}
