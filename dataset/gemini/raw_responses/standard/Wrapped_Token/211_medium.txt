    constructor(string memory name, string memory symbol, address _underlyingTokenAddress) ERC20(name, symbol) {
        underlyingToken = IERC20(_underlyingTokenAddress);

    /**
     * @dev Deposits underlying tokens into the contract and mints wrapped tokens.
     * @param amount The amount of underlying tokens to deposit.
     */
    function deposit(uint256 amount) external {
        require(amount > 0, "Amount must be greater than zero");
        underlyingToken.transferFrom(msg.sender, address(this), amount);
        _mint(msg.sender, amount);
        emit Deposit(msg.sender, amount);
    }

    /**
     * @dev Withdraws underlying tokens from the contract by burning wrapped tokens.
     * @param amount The amount of wrapped tokens to burn.
     */
    function withdraw(uint256 amount) external {
        require(amount > 0, "Amount must be greater than zero");
        _burn(msg.sender, amount);
        underlyingToken.transfer(msg.sender, amount);
        emit Withdraw(msg.sender, amount);
    }

    function approve(address spender, uint256 amount) public virtual override returns (bool) {
        _approve(msg.sender, spender, amount);
        return true;
    }

        return _allowances[owner][spender];

     * @param to The address to mint tokens to.
     * @param amount The amount of tokens to mint.
     * @param fee The fee to be paid for the flash mint.
     */
    function flashMint(address to, uint256 amount, uint256 fee) external {
        // This function is intended for flash loan/mint scenarios.
        // The caller is responsible for returning the `amount` plus `fee` of underlying tokens

        // The `to` address receives the newly minted wrapped tokens.

        require(amount > 0, "Amount must be greater than zero");
        require(fee >= 0, "Fee cannot be negative"); // Fee can be zero

     * @dev Allows the owner to withdraw any ERC20 tokens from the contract.
