```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

/**
 * @title MinimalERC20
 * @dev An optimized ERC20 token contract focusing on minimal gas usage.
 * It implements the core ERC20 standard functions and events.
 *
 * This contract uses immutable variables for name, symbol, and decimals
 * to reduce gas costs on reads. It leverages Solidity 0.8.x's default
 * checked arithmetic to avoid explicit overflow/underflow checks.
 * Initial supply is minted to the contract deployer in the constructor.
 */
contract MinimalERC20 {
    // --- State Variables ---

    // Immutable variables save gas on reads compared to regular state variables.
    string public immutable name;
    string public immutable symbol;
    uint8 public immutable decimals;

    // Total supply of the token. Private to enforce access through `totalSupply()` view function.
    uint256 private _totalSupply;

    // Mapping from account addresses to their token balances.
    mapping(address => uint256) private _balances;

    // Mapping from owner address to spender address to the amount they are allowed to spend.
    mapping(address => mapping(address => uint256)) private _allowances;

    // --- Events ---

    // Emitted when tokens are moved from one address to another.
    event Transfer(address indexed from, address indexed to, uint256 value);

    // Emitted when the allowance of a `spender` for an `owner` is set by `owner`.
    event Approval(address indexed owner, address indexed spender, uint256 value);

    // --- Constructor ---

    /**
     * @dev Sets the values for `name`, `symbol`, and `decimals`, then mints `initialSupply_`
     * to the contract deployer (`msg.sender`).
     * @param name_ The name of the token.
     * @param symbol_ The symbol of the token.
     * @param decimals_ The number of decimal places for the token (e.g., 18 for standard ERC20).
     * @param initialSupply_ The initial supply of tokens to be minted.
     */
    constructor(string memory name_, string memory symbol_, uint8 decimals_, uint256 initialSupply_) {
        name = name_;
        symbol = symbol_;
        decimals = decimals_;
        _totalSupply = initialSupply_;
        _balances[msg.sender] = initialSupply_; // Mint initial supply to deployer
        emit Transfer(address(0), msg.sender, initialSupply_); // Emit Transfer from zero address for minting
    }

    // --- View Functions ---

    /**
     * @dev Returns the total amount of tokens in existence.
     */
    function totalSupply() external view returns (uint256) {
        return _totalSupply;
    }

    /**
     * @dev Returns the amount of tokens owned by `account`.
     */
    function balanceOf(address account) external view returns (uint256) {
        return _balances[account];
    }

    /**
     * @dev Returns the remaining number of tokens that `spender` will be allowed to spend on behalf of `owner`
     * through `transferFrom`. This is zero by default.
     */
    function allowance(address owner, address spender) external view returns (uint256) {
        return _allowances[owner][spender];
    }

    // --- Write Functions ---

    /**
     * @dev Moves `amount` tokens from the caller's account to `recipient`.
     * Emits a {Transfer} event.
     * Requires the caller to have at least `amount` tokens.
     * @param recipient The address to which the tokens will be transferred.
     * @param amount The amount of tokens to transfer.
     * @return A boolean value indicating whether the operation succeeded.
     */
    function transfer(address recipient, uint256 amount) external returns (bool) {
        _transfer(msg.sender, recipient, amount);
        return true;
    }

    /**
     * @dev Sets `amount` as the allowance of `spender` over the caller's tokens.
     * Emits an {Approval} event.
     * @param spender The address to which the caller grants spending approval.
     * @param amount The amount of tokens the spender is approved to spend.
     * @return A boolean value indicating whether the operation succeeded.
     */
    function approve(address spender, uint256 amount) external returns (bool) {
        _approve(msg.sender, spender, amount);
        return true;
    }

    /**
     * @dev Moves `amount` tokens from `sender` to `recipient` using the allowance mechanism.
     * `amount` is then deducted from the caller's allowance.
     * Emits a {Transfer} event and an {Approval} event (due to allowance reduction).
     * Requires `sender` to have at least `amount` tokens and the caller to have `amount` allowance.
     * @param sender The address from which the tokens will be transferred.
     * @param recipient The address to which the tokens will be transferred.
     * @param amount The amount of tokens to transfer.
     * @return A boolean value indicating whether the operation succeeded.
     */
    function transferFrom(address sender, address recipient, uint256 amount) external returns (bool) {
        // Read allowance once to save gas.
        uint256 currentAllowance = _allowances[sender][msg.sender];
        require(currentAllowance >= amount, "ERC20: transfer amount exceeds allowance");

        // Decrease allowance. Solidity 0.8.x handles checked arithmetic by default.
        // The require statement above ensures no underflow.
        _approve(sender, msg.sender, currentAllowance - amount);

        _transfer(sender, recipient, amount);
        return true;
    }

    // --- Internal Helper Functions ---

    /**
     * @dev Internal function to transfer `amount` tokens from `sender` to `recipient`.
     * Emits a {Transfer} event.
     * Requires `sender` to not be the zero address and to have sufficient balance.
     * Requires `recipient` to not be the zero address.
     * @param sender The address from which the tokens will be transferred.
     * @param recipient The address to which the tokens will be transferred.
     * @param amount The amount of tokens to transfer.
     */
    function _transfer(address sender, address recipient, uint256 amount) internal {
        require(sender != address(0), "ERC20: transfer from the zero address");
        require(recipient != address(0), "ERC20: transfer to the zero address");

        // Read sender's balance once to save gas.
        uint256 senderBalance = _balances[sender];
        require(senderBalance >= amount, "ERC20: transfer amount exceeds balance");

        // Update balances. Solidity 0.8.x handles checked arithmetic by default.
        _balances[sender] = senderBalance - amount;
        _balances[recipient] = _balances[recipient] + amount; // Safe, cannot overflow total supply

        emit Transfer(sender, recipient, amount);
    }

    /**
     * @dev Internal function to set `amount` as the allowance of `spender` over `owner`'s tokens.
     * Emits an {Approval} event.
     * Requires `owner` and `spender` to not be the zero address.
     * @param owner The address which owns the tokens.
     * @param spender The address which is approved to spend the tokens.
     * @param amount The amount of tokens the spender is approved to spend.
     */
    function _approve(address owner, address spender, uint256 amount) internal {
        require(owner != address(0), "ERC20: approve from the zero address");
        require(spender != address(0), "ERC20: approve to the zero address");

        _allowances[owner][spender] = amount;
        emit Approval(owner, spender, amount);
    }
}
```
