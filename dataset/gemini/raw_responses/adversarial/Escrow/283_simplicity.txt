```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract Escrow {
    address public owner;
    address public buyer;
    address public seller;
    uint256 public depositAmount;
    bool public isDeposited;
    bool public isReleased;
    bool public isRefunded;

    event Deposited(address indexed buyer, uint256 amount);
    event Released(address indexed seller, uint256 amount);
    event Refunded(address indexed buyer, uint256 amount);

    modifier onlyOwner() {
        require(msg.sender == owner, "Only the owner can perform this action");
        _;
    }

    modifier onlyBuyer() {
        require(msg.sender == buyer, "Only the buyer can perform this action");
        _;
    }

    modifier onlySeller() {
        require(msg.sender == seller, "Only the seller can perform this action");
        _;
    }

    modifier notDeposited() {
        require(!isDeposited, "Funds have already been deposited");
        _;
    }

    modifier isDeposited() {
        require(isDeposited, "Funds have not been deposited yet");
        _;
    }

    modifier notReleasedOrRefunded() {
        require(!isReleased && !isRefunded, "Funds have already been released or refunded");
        _;
    }

    constructor(address _buyer, address _seller) {
        owner = msg.sender;
        buyer = _buyer;
        seller = _seller;
    }

    function deposit() external payable onlyBuyer notDeposited {
        require(msg.value > 0, "Deposit amount must be greater than zero");
        depositAmount = msg.value;
        isDeposited = true;
        emit Deposited(msg.sender, msg.value);
    }

    function releaseFunds() external onlyOwner isDeposited notReleasedOrRefunded {
        require(seller != address(0), "Seller address is not set");
        isReleased = true;
        (bool success, ) = payable(seller).call{value: address(this).balance}(
            ""
        );
        require(success, "Funds transfer failed");
        emit Released(seller, address(this).balance);
    }

    function refundFunds() external onlyOwner isDeposited notReleasedOrRefunded {
        require(buyer != address(0), "Buyer address is not set");
        isRefunded = true;
        (bool success, ) = payable(buyer).call{value: address(this).balance}(
            ""
        );
        require(success, "Funds transfer failed");
        emit Refunded(buyer, address(this).balance);
    }

    receive() external payable {}
}
```
