contract TestnetFlashLoanProvider {

    uint256 public flashLoanFeePercentage = 30; // 0.3%

    address payable public feeRecipient;

        feeRecipient = _feeRecipient;

     * @notice Calculates the amount to be repaid for a flash loan.
     * @param _amount The amount borrowed.
     * @return The total amount to be repaid (borrowed amount + fee).

    function getRepaymentAmount(uint256 _amount) public view returns (uint256) {
        uint256 fee = (_amount * flashLoanFeePercentage) / 10000; // 10000 for 0.01% precision
        return _amount + fee;

     * @notice Initiates a flash loan.

     * @param _amount The amount of the token to borrow.

    function requestFlashLoan(

        address _asset,
        uint256 _amount,
        address _flashLoanReceiver,
        bytes calldata _data

        IFlashLoanReceiver receiver = IFlashLoanReceiver(_flashLoanReceiver);
        receiver.executeOperation(_borrower, _asset, _amount, _data);

        // Calculate repayment amount (borrowed amount + fee)
        uint256 repaymentAmount = getRepaymentAmount(_amount);

        uint256 borrowerBalanceAfter = token.balanceOf(_borrower);

     * @notice This function is called by the flash loan receiver to repay the loan and fee.

    function repayFlashLoan(address _asset, uint256 _amount) external {

        uint256 borrowedAmount = _amount - fee;

        require(token.balanceOf(address(this)) >= _amount, "FlashLoanProvider: Insufficient repayment amount");

        if (fee > 0) {
            bool feeTransferSuccess = token.transfer(feeRecipient, fee);
            require(feeTransferSuccess, "FlashLoanProvider: Fee transfer failed");
        }

        emit LoanRepaid(msg.sender, _asset, _amount);
    }

         * @param _asset The address of the ERC20 token borrowed.
         * @param _amount The amount of the token borrowed.
         * @param _data Arbitrary data passed from the flash loan request.
         * @return A boolean indicating success or failure of the operation.
         */
        function executeOperation(
            address _borrower,
            address _asset,
            uint256 _amount,
            bytes calldata _data
        ) external returns (bool);
    }
