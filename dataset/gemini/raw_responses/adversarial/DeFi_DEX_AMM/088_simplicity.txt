```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract SimpleAMM is Ownable {
    IERC20 public tokenA;
    IERC20 public tokenB;

    uint256 public reserveA;
    uint256 public reserveB;

    constructor(address _tokenA, address _tokenB) {
        tokenA = IERC20(_tokenA);
        tokenB = IERC20(_tokenB);
    }

    function addLiquidity(uint256 amountA, uint256 amountB) external {
        require(amountA > 0 && amountB > 0, "Amounts must be positive");

        uint256 currentReserveA = reserveA;
        uint256 currentReserveB = reserveB;

        if (currentReserveA == 0 && currentReserveB == 0) {
            reserveA = amountA;
            reserveB = amountB;
        } else {
            uint256 requiredAmountA = (amountB * currentReserveA) / currentReserveB;
            uint256 requiredAmountB = (amountA * currentReserveB) / currentReserveA;

            if (amountA >= requiredAmountA) {
                reserveA = currentReserveA + amountA;
                reserveB = currentReserveB + amountB;
            } else {
                reserveA = currentReserveA + requiredAmountA;
                reserveB = currentReserveB + amountB;
            }
        }

        tokenA.transferFrom(msg.sender, address(this), amountA);
        tokenB.transferFrom(msg.sender, address(this), amountB);
    }

    function removeLiquidity(uint256 amountA, uint256 amountB) external {
        require(amountA > 0 && amountB > 0, "Amounts must be positive");
        require(amountA <= reserveA, "Insufficient reserve A");
        require(amountB <= reserveB, "Insufficient reserve B");

        reserveA -= amountA;
        reserveB -= amountB;

        tokenA.transfer(msg.sender, amountA);
        tokenB.transfer(msg.sender, amountB);
    }

    function swapExactAForB(uint256 amountInA, uint256 amountOutMinB) external returns (uint256 amountOutB) {
        require(amountInA > 0, "Amount in must be positive");

        uint256 currentReserveA = reserveA;
        uint256 currentReserveB = reserveB;

        uint256 amountOutB = (amountInA * currentReserveB) / (currentReserveA + amountInA);
        require(amountOutB >= amountOutMinB, "Insufficient output amount");

        reserveA = currentReserveA + amountInA;
        reserveB = currentReserveB - amountOutB;

        tokenA.transferFrom(msg.sender, address(this), amountInA);
        tokenB.transfer(msg.sender, amountOutB);

        return amountOutB;
    }

    function swapExactBForA(uint256 amountInB, uint256 amountOutMinA) external returns (uint256 amountOutA) {
        require(amountInB > 0, "Amount in must be positive");

        uint256 currentReserveA = reserveA;
        uint256 currentReserveB = reserveB;

        uint256 amountOutA = (amountInB * currentReserveA) / (currentReserveB + amountInB);
        require(amountOutA >= amountOutMinA, "Insufficient output amount");

        reserveB = currentReserveB + amountInB;
        reserveA = currentReserveA - amountOutA;

        tokenB.transferFrom(msg.sender, address(this), amountInB);
        tokenA.transfer(msg.sender, amountOutA);

        return amountOutA;
    }

    function getAmountOutB(uint256 amountInA) public view returns (uint256 amountOutB) {
        uint256 currentReserveA = reserveA;
        uint256 currentReserveB = reserveB;
        return (amountInA * currentReserveB) / (currentReserveA + amountInA);
    }

    function getAmountOutA(uint256 amountInB) public view returns (uint256 amountOutA) {
        uint256 currentReserveA = reserveA;
        uint256 currentReserveB = reserveB;
        return (amountInB * currentReserveA) / (currentReserveB + amountInB);
    }
}
```
