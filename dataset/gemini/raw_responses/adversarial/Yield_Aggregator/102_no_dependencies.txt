// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract StandaloneYieldAggregator {

    address public owner;
    uint256 public totalDeposited;
    mapping(address => uint256) public balances;

    event Deposit(address indexed user, uint256 amount);
    event Withdrawal(address indexed user, uint256 amount);
    event YieldGenerated(uint256 amount);

    constructor() {
        owner = msg.sender;
    }

    modifier onlyOwner() {
        require(msg.sender == owner, "Only the owner can perform this action");
        _;
    }

    // This is a placeholder for actual yield generation logic.
    // In a real-world scenario, this would interact with other DeFi protocols
    // or internal strategies to generate yield.
    // For this standalone example, it simply increases the total deposited amount.
    function generateYield() public onlyOwner {
        // Simulate yield generation. In a real protocol, this would involve
        // complex logic like depositing into lending protocols, staking, etc.
        // For demonstration, we'll just add a fixed amount to totalDeposited.
        uint256 simulatedYield = 10 ether; // Example yield amount
        totalDeposited += simulatedYield;
        emit YieldGenerated(simulatedYield);
    }

    function deposit() public payable {
        require(msg.value > 0, "Deposit amount must be greater than zero");
        balances[msg.sender] += msg.value;
        totalDeposited += msg.value;
        emit Deposit(msg.sender, msg.value);
    }

    function withdraw(uint256 _amount) public {
        require(_amount > 0, "Withdrawal amount must be greater than zero");
        uint256 userBalance = balances[msg.sender];
        require(userBalance >= _amount, "Insufficient balance for withdrawal");

        // Ensure there's enough in totalDeposited to cover the withdrawal
        // This is a simplified check. In a real aggregator, you'd need to consider
        // how much of totalDeposited is actually available for withdrawal after
        // accounting for yield generation and any fees.
        require(totalDeposited >= _amount, "Not enough total deposited funds to cover withdrawal");

        balances[msg.sender] -= _amount;
        totalDeposited -= _amount;

        (bool success, ) = payable(msg.sender).call{value: _amount}("");
        require(success, "Withdrawal failed");

        emit Withdrawal(msg.sender, _amount);
    }

    function withdrawAll() public {
        uint256 userBalance = balances[msg.sender];
        require(userBalance > 0, "No balance to withdraw");
        withdraw(userBalance);
    }

    // Function to claim yield, simulating distribution of generated yield.
    // In a real aggregator, this would involve calculating and distributing
    // the earned yield to depositors proportionally.
    // For this standalone example, we'll assume yield generation directly
    // increases `totalDeposited` and withdrawal logic handles proportional distribution.
    // This function is more of a placeholder to illustrate where yield claiming might occur.
    function claimYield() public {
        // In a real aggregator, this function would:
        // 1. Interact with yield-generating protocols to harvest rewards.
        // 2. Distribute these rewards to users based on their stake.
        // For this simplified standalone contract, the `generateYield` function
        // directly increases `totalDeposited`, and the withdrawal logic
        // implicitly handles proportional distribution of the *total* pool.
        // No explicit user-level yield claiming is implemented here for simplicity.
        // If you wanted to implement explicit yield claiming, you'd need:
        // - A way to track earned yield per user.
        // - A mechanism to deposit harvested yield into the contract.
        // - A function to allow users to claim their earned yield.
    }

    // Fallback function to receive Ether directly
    receive() external payable {
        deposit();
    }

    // Fallback function for calls that don't match any function
    fallback() external payable {
        deposit();
    }

    function getBalance(address _user) public view returns (uint256) {
        return balances[_user];
    }

    function getTotalDeposited() public view returns (uint256) {
        return totalDeposited;
    }
}
