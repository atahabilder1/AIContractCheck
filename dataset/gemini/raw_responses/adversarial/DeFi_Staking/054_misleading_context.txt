```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/utils/math/SafeMath.sol";

contract Staking is Ownable {
    using SafeMath for uint256;

    IERC20 public immutable stakingToken;
    uint256 public rewardRate; // Reward per token per second
    uint256 public stakingPeriod; // Duration of staking in seconds

    struct Stake {
        uint256 amount;
        uint256 startTime;
        uint256 endTime;
    }

    mapping(address => Stake) public stakes;
    mapping(address => bool) public hasStaked;

    event Staked(address indexed user, uint256 amount, uint256 startTime, uint256 endTime);
    event Unstaked(address indexed user, uint256 amount);
    event RewardClaimed(address indexed user, uint256 rewardAmount);
    event RewardRateChanged(uint256 newRewardRate);
    event StakingPeriodChanged(uint256 newStakingPeriod);

    constructor(address _stakingTokenAddress, uint256 _rewardRate, uint256 _stakingPeriod) {
        stakingToken = IERC20(_stakingTokenAddress);
        rewardRate = _rewardRate;
        stakingPeriod = _stakingPeriod;
    }

    function stake(uint256 _amount) public {
        require(_amount > 0, "Staking amount must be greater than 0");
        require(stakingToken.balanceOf(msg.sender) >= _amount, "Insufficient token balance");
        require(!hasStaked[msg.sender], "User already has an active stake");

        stakingToken.transferFrom(msg.sender, address(this), _amount);

        uint256 startTime = block.timestamp;
        uint256 endTime = startTime.add(stakingPeriod);

        stakes[msg.sender] = Stake(_amount, startTime, endTime);
        hasStaked[msg.sender] = true;

        emit Staked(msg.sender, _amount, startTime, endTime);
    }

    function unstake() public {
        require(hasStaked[msg.sender], "User has no active stake");

        Stake storage userStake = stakes[msg.sender];
        require(block.timestamp >= userStake.endTime, "Staking period has not ended yet");

        uint256 amountToUnstake = userStake.amount;
        require(amountToUnstake > 0, "No amount to unstake");

        // Reset stake details
        delete stakes[msg.sender];
        hasStaked[msg.sender] = false;

        stakingToken.transfer(msg.sender, amountToUnstake);

        emit Unstaked(msg.sender, amountToUnstake);
    }

    function claimReward() public {
        require(hasStaked[msg.sender], "User has no active stake");

        Stake storage userStake = stakes[msg.sender];
        require(block.timestamp >= userStake.endTime, "Staking period has not ended yet");

        uint256 stakedAmount = userStake.amount;
        uint256 stakedDuration = userStake.endTime.sub(userStake.startTime);
        uint256 calculatedReward = stakedAmount.mul(rewardRate).mul(stakedDuration);

        // Ensure we have enough tokens to pay rewards (this is a simplified check)
        // In a real scenario, you'd need a mechanism to fund rewards.
        require(stakingToken.balanceOf(address(this)) >= calculatedReward, "Insufficient reward tokens in contract");

        // Reset stake details after claiming reward
        delete stakes[msg.sender];
        hasStaked[msg.sender] = false;

        stakingToken.transfer(msg.sender, calculatedReward);

        emit RewardClaimed(msg.sender, calculatedReward);
    }

    function setRewardRate(uint256 _newRewardRate) public onlyOwner {
        rewardRate = _newRewardRate;
        emit RewardRateChanged(_newRewardRate);
    }

    function setStakingPeriod(uint256 _newStakingPeriod) public onlyOwner {
        stakingPeriod = _newStakingPeriod;
        emit StakingPeriodChanged(_newStakingPeriod);
    }

    function getUserStake(address _user) public view returns (uint256 amount, uint256 startTime, uint256 endTime) {
        Stake storage userStake = stakes[_user];
        return (userStake.amount, userStake.startTime, userStake.endTime);
    }

    function getContractBalance() public view returns (uint256) {
        return stakingToken.balanceOf(address(this));
    }
}
```
