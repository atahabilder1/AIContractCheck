pragma solidity ^0.8.20;

import {IL2Bridge, IL1Bridge} from "./interfaces.sol";

contract CrossChainMessenger {
    // --- State Variables ---
    address public immutable L1_BRIDGE;
    address public immutable L2_BRIDGE;

    // --- Events ---
    event MessageSent(uint256 indexed messageId, address indexed from, bytes message);
    event MessageReceived(uint256 indexed messageId, address indexed from, bytes message);

    // --- Constructor ---
    constructor(address l1Bridge, address l2Bridge) {
        L1_BRIDGE = l1Bridge;
        L2_BRIDGE = l2Bridge;
    }

    // --- Public Functions ---

    /**
     * @notice Sends a message from L1 to L2.
     * @param _message The message payload to send.
     */
    function sendMessageL1ToL2(bytes calldata _message) external {
        // Revert if called on L2
        // Note: This check is crucial for security and gas efficiency.
        // If this function is called on L2, it's an error and we want to fail fast.
        if (address(this) == L2_BRIDGE) {
            revert("CrossChainMessenger: Cannot send L1->L2 message from L2");
        }

        // Directly call the L1 bridge to send the message.
        // Assumes the L1 bridge has a function like `sendMessage(bytes calldata message)`
        // and that this contract is authorized to call it.
        IL1Bridge(L1_BRIDGE).sendMessage(_message);

        // Emit an event for tracking.
        // The messageId would typically be generated by the bridge contract.
        // For simplicity, we'll use a placeholder or assume the bridge returns it.
        // In a real-world scenario, you'd likely get this from the bridge's return value or event.
        emit MessageSent(block.number, msg.sender, _message); // Placeholder messageId
    }

    /**
     * @notice Sends a message from L2 to L1.
     * @param _message The message payload to send.
     */
    function sendMessageL2ToL1(bytes calldata _message) external {
        // Revert if called on L1
        if (address(this) == L1_BRIDGE) {
            revert("CrossChainMessenger: Cannot send L2->L1 message from L1");
        }

        // Directly call the L2 bridge to send the message.
        // Assumes the L2 bridge has a function like `sendMessage(bytes calldata message)`
        // and that this contract is authorized to call it.
        IL2Bridge(L2_BRIDGE).sendMessage(_message);

        // Emit an event for tracking.
        emit MessageSent(block.number, msg.sender, _message); // Placeholder messageId
    }

    /**
     * @notice Receives a message from L1. This function is typically called by the L1 bridge.
     * @param _messageId The unique identifier for the message.
     * @param _from The address on L1 that sent the message.
     * @param _message The message payload.
     */
    function receiveMessageFromL1(uint256 _messageId, address _from, bytes calldata _message) external {
        // Revert if not called by the L1 bridge.
        // This is a critical security check to ensure only the designated bridge can relay messages.
        if (msg.sender != L1_BRIDGE) {
            revert("CrossChainMessenger: Invalid sender for receiveMessageFromL1");
        }

        // Revert if called on L1 (this contract should only be deployed on L2 for this function)
        if (address(this) == L1_BRIDGE) {
            revert("CrossChainMessenger: Cannot receive L1 message on L1 bridge");
        }

        // Process the received message.
        // In a real application, you would implement logic here to handle the message,
        // e.g., call another contract function, update state, etc.
        // For gas optimization, we avoid unnecessary storage writes if the message
        // doesn't require state changes.

        emit MessageReceived(_messageId, _from, _message);
    }

    /**
     * @notice Receives a message from L2. This function is typically called by the L2 bridge.
     * @param _messageId The unique identifier for the message.
     * @param _from The address on L2 that sent the message.
     * @param _message The message payload.
     */
    function receiveMessageFromL2(uint256 _messageId, address _from, bytes calldata _message) external {
        // Revert if not called by the L2 bridge.
        if (msg.sender != L2_BRIDGE) {
            revert("CrossChainMessenger: Invalid sender for receiveMessageFromL2");
        }

        // Revert if called on L2 (this contract should only be deployed on L1 for this function)
        if (address(this) == L2_BRIDGE) {
            revert("CrossChainMessenger: Cannot receive L2 message on L2 bridge");
        }

        // Process the received message.
        emit MessageReceived(_messageId, _from, _message);
    }

    // --- Internal/Private Functions ---
    // No additional internal/private functions are strictly necessary for basic messaging,
    // keeping the contract lean.

    // --- Fallback/Receive Functions ---
    // No fallback or receive functions are needed for this specific contract's functionality,
    // as all interactions are through explicit function calls.
}
